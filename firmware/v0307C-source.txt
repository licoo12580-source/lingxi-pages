static void diagnosePartitions() {
    LOG_I("====== DIAG: Flash信息 ======");
    // 1. Flash总大小
    uint32_t flashSize = ESP.getFlashChipSize();
    LOG_I("FlashSize: %uMB (%u)", flashSize / 1048576, flashSize);
    // 2. PSRAM
#if CONFIG_SPIRAM
    LOG_I("PSRAM: %uMB total, %uk free", ESP.getPsramSize() / 1048576, ESP.getFreePsram() / 1024);
#else
    LOG_I("PSRAM: 未启用");
#endif
    // 3. 枚举所有APP分区
    LOG_I("APP分区:");
    esp_partition_iterator_t it = esp_partition_find(ESP_PARTITION_TYPE_APP, ESP_PARTITION_SUBTYPE_ANY, NULL);
    int appCount = 0;
    for (; it; it = esp_partition_next(it)) {
        const esp_partition_t* p = esp_partition_get(it);
        LOG_I("  [%d] %s @0x%x sz=%uKB(%uMB)",
            appCount++, p->label, p->address, p->size / 1024, p->size / 1048576);
    }
    if (it) esp_partition_iterator_release(it);
    // 4. 运行中的分区
    const esp_partition_t* rp = esp_ota_get_running_partition();
    if (rp) {
        LOG_I("运行分区: %s @0x%x sz=%uKB", rp->label, rp->address, rp->size / 1024);
        esp_ota_img_states_t st;
        if (esp_ota_get_state_partition(rp, &st) == ESP_OK) {
            LOG_I("  状态: %s",
                st == ESP_OTA_IMG_VALID ? "VALID" :
                st == ESP_OTA_IMG_NEW ? "NEW" :
                st == ESP_OTA_IMG_PENDING_VERIFY ? "PENDING_VERIFY" :
                st == ESP_OTA_IMG_UNDEFINED ? "UNDEFINED" :
                st == ESP_OTA_IMG_INVALID ? "INVALID" :
                st == ESP_OTA_IMG_ABORTED ? "ABORTED" : "未知");
        }
    }
    // 5. 下一个OTA分区
    const esp_partition_t* np = esp_ota_get_next_update_partition(NULL);
    if (np) {
        LOG_I("OTA更新分区: %s @0x%x sz=%uKB", np->label, np->address, np->size / 1024);
    } else {
        LOG_E("OTA更新分区: 不存在! 分区表没有OTA槽");
    }
    // 6. 枚举DATA分区（otadata等）
    it = esp_partition_find(ESP_PARTITION_TYPE_DATA, ESP_PARTITION_SUBTYPE_ANY, NULL);
    LOG_I("DATA分区:");
    for (; it; it = esp_partition_next(it)) {
        const esp_partition_t* p = esp_partition_get(it);
        LOG_I("  %s @0x%x sz=%uKB subtype=0x%x", p->label, p->address, p->size / 1024, p->subtype);
    }
    if (it) esp_partition_iterator_release(it);
    LOG_I("====== DIAG End ======");
}

static void sendFrame(uint8_t t, const uint8_t* d, uint16_t l) {
    uint8_t h[5] = {FS1, FS2, t, (uint8_t)(l & 0xFF), (uint8_t)(l >> 8)};
    uint8_t c = FS1 ^ FS2 ^ t ^ (l & 0xFF) ^ (l >> 8);
    for (uint16_t i = 0; i < l; i++) c ^= d[i];
    Serial1.write(h, 5); Serial1.write(d, l); Serial1.write(c);
}

static unsigned long lastMon = 0;
void monSystem() {
    if (millis() - lastMon < 10000) return;
    lastMon = millis();
    static UBaseType_t lastStack = 0;
    UBaseType_t stack = uxTaskGetStackHighWaterMark(NULL);
    size_t freeHeap = heap_caps_get_free_size(MALLOC_CAP_INTERNAL);
    LOG_I("MEM: heap=%u stack=%u", freeHeap, stack);
    if (stack < lastStack - 256 && lastStack) LOG_E("Stack shrink: %u->%u", lastStack, stack);
    if (freeHeap < 20000) LOG_E("Low heap: %u", freeHeap);
    lastStack = stack;
    static int status_cnt = 0;
    if (++status_cnt >= 6) {  // ~60秒
        status_cnt = 0;
        Serial.printf("<v0307C pp=%d heap=%uk\n", pepCount, freeHeap/1024);
    }
}

// ====== 版本回滚 ======
void rollbackFirmware() {
    const esp_partition_t* running = esp_ota_get_running_partition();
    const esp_partition_t* next = esp_ota_get_next_update_partition(NULL);
    if (next && running->subtype != next->subtype) {
        LOG_I("Rollback to backup partition");
        esp_ota_set_boot_partition(next);
        delay(500);
        ESP.restart();
    } else {
        LOG_E("No backup partition to rollback");
    }
}

// ====== 流式OTA（统一使用IDF原生API） ======
void startStreamOta(size_t comp_sz, size_t orig_sz) {
    ota.state = OTA_RECEIVE; ota.comp_size = comp_sz; ota.comp_received = 0;
    ota.decomp_written = 0; ota.timeout = millis() + OTA_TIMEOUT_MS; ota.sha256[0] = 0;
    tinfl_init(&ota.decomp);
    // 获取OTA更新分区
    ota_part = esp_ota_get_next_update_partition(NULL);
    if (!ota_part) {
        reportError("NO_OTA_PART", "分区表没有OTA槽");
        ota.state = OTA_ERROR; return;
    }
    if (ota_part->size < orig_sz) {
        char b[64]; snprintf(b,64,"分区%uKB<固件%uKB", ota_part->size/1024, (unsigned)(orig_sz/1024));
        reportError("PART_TOO_SMALL", b);
        ota.state = OTA_ERROR; return;
    }
    // 开始OTA会话
    esp_err_t err = esp_ota_begin(ota_part, OTA_SIZE_UNKNOWN, &ota_handle);
    if (err != ESP_OK) {
        char b[64]; snprintf(b,64,"esp_ota_begin: %s", esp_err_to_name(err));
        reportError("OTA_BEGIN_FAIL", b);
        ota.state = OTA_ERROR; return;
    }
    LOG_I("OTA start: comp=%zu orig=%zu -> %s @0x%x sz=%zuKB",
        comp_sz, orig_sz, ota_part->label, ota_part->address, ota_part->size/1024);
    showMsg("OTA 接收中...");
}

void feedOTA(const uint8_t* data, size_t len) {
    if (ota.state != OTA_RECEIVE) return;
    const uint8_t* in = data; size_t in_remain = len;
    while (in_remain > 0) {
        uint8_t out_buf[TINFL_OUT_BUF_SIZE];
        size_t in_consumed = in_remain, out_produced = TINFL_OUT_BUF_SIZE;
        tinfl_status status = tinfl_decompress(&ota.decomp, in, &in_consumed, out_buf, out_buf, &out_produced, TINFL_FLAG_PARSE_ZLIB_HEADER | TINFL_FLAG_HAS_MORE_INPUT);
        ota.comp_received += in_consumed; in += in_consumed; in_remain -= in_consumed;
        if (out_produced > 0) {
            esp_err_t werr = esp_ota_write(ota_handle, out_buf, out_produced);
            if (werr != ESP_OK) {
                LOG_E("Flash write fail: %s", esp_err_to_name(werr));
                ota.state = OTA_ERROR; break;
            }
            ota.decomp_written += out_produced;
        }
        if (status == TINFL_STATUS_DONE) { LOG_I("Decompress done"); ota.state = OTA_VERIFY; break; }
        if (status == TINFL_STATUS_NEEDS_MORE_INPUT) break;
        if (status == TINFL_STATUS_HAS_MORE_OUTPUT) continue;
        if (status < 0) { LOG_E("Decompress fail: %d", status); ota.state = OTA_ERROR; break; }
    }
    if (ota.comp_size > 0) {
        int pct = ota.comp_received * 100 / ota.comp_size;
        static int lastPct = -1;
        if (pct >= lastPct + 5) { lastPct = pct; Serial1.printf("!P:%d%%\n", pct); showOtaProgress(pct); }
    }
}

// ====== 统一UART解析器 ======
String cmdBuf; uint8_t frameBuf[8192]; int framePos = 0; bool inBinary = false;

void parseUART() {
    static unsigned long lastDiag = 0;
    unsigned long now = millis();
    int bytesRead = 0;
    while (Serial1.available() && bytesRead < 8192) {
        uint8_t b = Serial1.read(); bytesRead++;
        if (!inBinary) {
            if (b == '\n' || b == '\r') { if (cmdBuf.length() > 0) { procCmd(cmdBuf); cmdBuf = ""; } }
            else if (cmdBuf.length() < 256) cmdBuf += (char)b;
            if (framePos == 0 && b == FS1) { frameBuf[0] = FS1; framePos = 1; }
            else if (framePos == 1 && b == FS2) { frameBuf[1] = FS2; framePos = 2; inBinary = true; }
            else framePos = 0;
        } else {
            if ((uint32_t)framePos >= sizeof(frameBuf)) {
                reportError("FRAME_OFLO", "帧缓冲溢出");
                framePos = 0; inBinary = false; continue;
            }
            frameBuf[framePos++] = b;
            if (framePos >= 5) {
                uint32_t plen = (frameBuf[4] << 8) | frameBuf[3];
                uint32_t expected = 5 + plen + 1;
                if (expected > sizeof(frameBuf)) { framePos = 0; inBinary = false; continue; }
                if ((uint32_t)framePos == expected) {
                    uint8_t chk = FS1 ^ FS2 ^ frameBuf[2] ^ frameBuf[3] ^ frameBuf[4];
                    for (int i = 5; i < framePos - 1; i++) chk ^= frameBuf[i];
                    if (chk == frameBuf[framePos - 1]) {
                        if (frameBuf[2] == FT_GZDATA && ota.state == OTA_RECEIVE) {
                            if (rawOtaMode) {
                                esp_err_t werr = esp_ota_write(ota_handle, frameBuf + 5, plen);
                                if (werr != ESP_OK) {
                                    char b[64]; snprintf(b,64,"Flash写: %s", esp_err_to_name(werr));
                                    reportError("FLASH_WRITE", b); ota.state = OTA_ERROR;
                                } else {
                                    ota.comp_received += plen;
                                }
                            } else {
                                feedOTA(frameBuf + 5, plen);
                            }
                            uint8_t ack[] = {FS1, FS2, FT_ACK, 0, 0, 0xCC};
                            Serial1.write(ack, 6);
                        } else if (frameBuf[2] == FT_GZFINISH) {
                            LOG_I("OTA收到FINISH帧");
                            ota.state = OTA_VERIFY;
                        } else if (frameBuf[2] != FT_ACK) {
                            char b[64]; snprintf(b,64,"未知帧类型=0x%02x", frameBuf[2]);
                            reportError("UNKNOWN_FRAME", b);
                        }
                    } else {
                        char b[64]; snprintf(b,64,"CRC校验: 计算0x%02x!=收到0x%02x", chk, frameBuf[framePos-1]);
                        reportError("CRC_FAIL", b);
                    }
                    framePos = 0; inBinary = false;
                }
            }
        }
    }
    if (bytesRead > 0 && now - lastDiag > 1000) {
        lastDiag = now;
        LOG_I("UART: 收到%d字节, ota.state=%d, inBinary=%d", bytesRead, ota.state, inBinary?1:0);
    }
}

void processOTA() {
    if (millis() > ota.timeout) {
        reportError("OTA_TIMEOUT", "超时，OTA取消");
        if(ota_handle){esp_ota_abort(ota_handle);ota_handle=0;}
        ota.state = OTA_IDLE; return;
    }
    parseUART();
    if (ota.state == OTA_VERIFY) {
        // 第一步：先结束OTA会话，确保固件完全写入Flash
        esp_err_t err = esp_ota_end(ota_handle);
        if (err != ESP_OK) {
            char b[64]; snprintf(b,64,"esp_ota_end: %s", esp_err_to_name(err));
            reportError("OTA_END_FAIL", b);
            ota.state = OTA_ERROR;
        } else {
            LOG_I("OTA结束成功");
            // 第二步：SHA256校验（必须在esp_ota_end之后，固件才完全可读）
            if (ota.sha256[0]) {
                LOG_I("SHA256: 期望=%s", ota.sha256);
                uint8_t calc_hash[32]; char hash_str[65];
                mbedtls_sha256_context ctx;
                mbedtls_sha256_init(&ctx); mbedtls_sha256_starts(&ctx, 0);
                uint8_t* buf = (uint8_t*)malloc(4096);
                if (buf) {
                    for (size_t off = 0; off < ota_part->size; off += 4096) {
                        size_t rd = (ota_part->size - off < 4096) ? ota_part->size - off : 4096;
                        if (esp_partition_read(ota_part, off, buf, rd) == ESP_OK)
                            mbedtls_sha256_update(&ctx, buf, rd);
                    }
                    free(buf);
                }
                mbedtls_sha256_finish(&ctx, calc_hash); mbedtls_sha256_free(&ctx);
                for (int i = 0; i < 32; i++) sprintf(hash_str + i*2, "%02x", calc_hash[i]);
                hash_str[64] = 0;
                LOG_I("SHA256: 计算=%s", hash_str);
                if (strcasecmp(hash_str, ota.sha256) != 0) {
                    reportError("SHA256_MISMATCH", "校验不匹配");
                    ota.state = OTA_ERROR; return;
                }
                LOG_I("SHA256通过✅");
            }
            // 第三步：设置启动分区
            err = esp_ota_set_boot_partition(ota_part);
            if (err != ESP_OK) {
                char b[64]; snprintf(b,64,"set_boot: %s", esp_err_to_name(err));
                reportError("SET_BOOT_FAIL", b);
                ota.state = OTA_ERROR;
            } else {
                LOG_I("OTA done, reboot...");
                Serial1.println("!STATUS:DONE");
                showMsg("OTA完成!");
                delay(1000);
                ESP.restart();
            }
        }
    }
    if (ota.state == OTA_ERROR) {
        rawOtaMode = false;
        reportError("OTA_FAIL", "OTA失败，即将重启");
        delay(500);
        ESP.restart();
    }
}

// ====== UART OTA处理（!ESP32OTA:模式，使用IDF原生API） ======
void processUartOta() {
    int processed = 0;
    while (Serial1.available() && uartOtaReceived < uartOtaSize && processed < 4096) {
        uint8_t b = Serial1.read();
        uartOtaBuf[uartOtaIdx++] = b;
        uartOtaReceived++;
        processed++;
        uartOtaTimeout = millis() + OTA_TIMEOUT_MS;
        if (uartOtaIdx >= 512 || uartOtaReceived >= uartOtaSize) {
            esp_ota_write(ota_handle, uartOtaBuf, uartOtaIdx);
            uartOtaIdx = 0;
            delay(3);
        }
        if (uartOtaSize > 0) {
            static int lastPct = -1;
            int pct = uartOtaReceived * 100 / uartOtaSize;
            if (pct != lastPct && pct % 5 == 0) {
                lastPct = pct;
                Serial.printf("[UOTA] %d%%\n", pct);
                Serial1.printf("!P:%d%%\n", pct);
                showOtaProgress(pct);
            }
        }
    }
    if (millis() > uartOtaTimeout) { esp_ota_abort(ota_handle); uartOtaMode = false; reportError("UOTA_TIMEOUT", "UOTA超时"); }
    if (uartOtaReceived >= uartOtaSize && !uartOtaDone) {
        uartOtaDone = true;
        showOtaProgress(100);
        esp_err_t err = esp_ota_end(ota_handle);
        if (err == ESP_OK) {
            err = esp_ota_set_boot_partition(ota_part);
            if (err == ESP_OK) {
                LOG_I("UOTA完成"); Serial1.println("!STATUS:DONE"); showMsg("UOTA完成!"); delay(500); ESP.restart();
            }
        }
        char b[64]; snprintf(b,64,"UOTA: %s", esp_err_to_name(err));
        reportError("UOTA_FAIL", b);
        delay(1000); ESP.restart();
    }
}

void procCmd(const String& c) {
    LOG_I("CMD: %s", c.c_str());
#ifdef USE_ML307C
    if (c == "!ML307") { ml307_passthru(); return; }
    if (c == "!OTA") { ml307_ota(); return; }
    if (c == "!DIAG") {
        ml307_diag(); return;
    }
    // I2S引脚配置 (I2S:WS=14,BCK=13,SD=12)
    if (c.startsWith("I2S:WS=")) {
        int w,b,s;
        if (sscanf(c.c_str(), "I2S:WS=%d,BCK=%d,SD=%d", &w, &b, &s) == 3) {
            savePinCfg(w,b,s); delay(1000); ESP.restart();
        }
        return;
    }
    // MQTT远程命令(从xiaobing/cmd订阅消息)
    if (c.startsWith("+MQTTURC:\"publish\"")) {
        if (c.indexOf("xiaobing/cmd") >= 0) {
            if (c.indexOf("OTA") >= 0) { LOG_I("收到远程OTA命令"); ml307_ota(); return; }
        }
    }
#endif
    // !UART:频率切换
    int uri = c.indexOf("!UART:");
    if (uri >= 0) {
        unsigned long r = c.substring(uri + 6).toInt();
        if (r >= 115200 && r <= 2000000) {
            LOG_I("UART switch: %lu", r);
            Serial1.printf("!UART:OK\n");
            delay(50);
            Serial1.flush();
            Serial1.updateBaudRate(r);
            LOG_I("UART switched to %lu", r);
        }
        return;
    }
    // !ESP32OTA:旧版UART OTA（v1.47兼容）
    int uoi = c.indexOf("!ESP32OTA:");
    if (uoi >= 0) {
        // 🔥 清空Serial1接收缓冲区，避免垃圾数据混入固件
        Serial1.flush();
        delay(10);
        while (Serial1.available()) Serial1.read();

        uartOtaSize = c.substring(uoi + 10).toInt();
        if (uartOtaSize > 100000 && uartOtaSize < 2097152) {
            uartOtaMode = true; uartOtaDone = false; uartOtaReceived = 0; uartOtaIdx = 0;
            uartOtaTimeout = millis() + OTA_TIMEOUT_MS;
            ota_part = esp_ota_get_next_update_partition(NULL);
            if (!ota_part) {
                LOG_E("UOTA begin FAILED: esp_ota_get_next_update_partition NULL");
                Serial1.println("!ERR:NO_OTA_PART"); uartOtaMode = false; return;
            }
            if ((size_t)uartOtaSize > ota_part->size) {
                LOG_E("UOTA begin FAILED: size %u > partition %u", uartOtaSize, ota_part->size);
                Serial1.println("!ERR:TOO_BIG"); uartOtaMode = false; return;
            }
            esp_err_t err = esp_ota_begin(ota_part, OTA_SIZE_UNKNOWN, &ota_handle);
            if (err != ESP_OK) {
                LOG_E("UOTA begin FAILED: esp_ota_begin %s", esp_err_to_name(err));
                Serial1.println("!ERR:OTA_BEGIN_FAIL"); uartOtaMode = false; return;
            }
            LOG_I("UOTA recv %u -> %s @0x%x", uartOtaSize, ota_part->label, ota_part->address);
        }
        return;
    }
    int ri = c.indexOf("RAWOTA:");
    if (ri >= 0) { size_t sz = c.substring(ri + 7).toInt(); if(sz > 100000) { rawOtaMode = true; startStreamOta(sz, sz); } return; }
    int oi = c.indexOf("GZOTA:");
    if (oi >= 0) { rawOtaMode = false; size_t comp = 0, orig = 0; sscanf(c.c_str() + oi + 6, "%zu:%zu", &comp, &orig); if (comp > 10000 && orig > 10000) startStreamOta(comp, orig); return; }
    if (c.startsWith("CRC:")) { if (ota.state == OTA_RECEIVE) { strncpy(ota.sha256, c.c_str() + 4, 64); ota.sha256[64] = 0; ota.state = OTA_VERIFY; } return; }
    if (c.startsWith("ir:")) {
        int c1 = c.indexOf(':', 3);
        String rest = c.substring(3);
        int colon = rest.indexOf(':');
        String brand = colon > 0 ? rest.substring(0, colon) : "";
        String param = colon > 0 ? rest.substring(colon + 1) : rest;
        if (brand == "" || brand == "nec") {
            // NEC format: ir:addr:code or ir:nec:addr:code
            if (colon > 0) {
                uint16_t addr = (uint16_t)rest.substring(0, colon).toInt();
                uint16_t code = (uint16_t)rest.substring(colon + 1).toInt();
                irsend.sendNEC(addr, code);
            } else {
                uint16_t addr = (uint16_t)param.toInt();
                irsend.sendNEC(addr, 0);
            }
            Serial1.println("{\"ir\":\"nec\"}");
        } else if (brand == "learn") {
            irrecv.resume();
            Serial1.println("{\"ir\":\"learn\"}");
        } else {
            Serial1.println("{\"ir\":\"unknown\"}");
        }
    }
    if (c.startsWith("irlearn:")) {
        irrecv.resume();
        Serial1.println("{\"ir\":\"learning\"}");
    }
    else if (c.startsWith("audio:")) { audioOn = (c[6] == '1'); if(audioOn) LOG_I("AUDIO ON"); else LOG_I("AUDIO OFF"); }
    else if (c == "!AUDIO:DIAG") {
        audioOn = true; audioDiagMode = 1; audioDiagCount = 0;
        audioDiagMin = 999; audioDiagMax = 0; audioDiagSum = 0;
        audioDiagStart = millis(); audioVolume = 0;
        if (!i2sOK) {
            LOG_I("I2S未就绪，尝试重新初始化...");
            setupI2S();
        }
        LOG_I("AUDIO DIAG START 30s (i2sOK=%d)", i2sOK?1:0);
        Serial1.printf("{\"diag\":\"start\",\"i2s\":%d,\"msg\":\"麦克风诊断30秒...\"}\n", i2sOK?1:0);
        if (!i2sOK) {
            LOG_E("I2S初始化失败，诊断无法进行");
            Serial1.println("{\"diag\":\"fail\",\"reason\":\"I2S初始化失败，检查硬件接线:WS=41 BCK=42 SD=2\"}");
            audioDiagMode = 0; audioOn = false;
        }
    }
    else if (c == "ping") { Serial1.println("{\"pong\":true}"); }
    else if (c.startsWith("asr:")) {
        asrText = c.substring(4);
        asrTextTime = millis();
    }
    else if (c == "status") { char b[128]; snprintf(b, sizeof(b), "{\"heap\":%u,\"ota\":%d,\"db\":%.0f,\"asr\":\"%s\",\"audio\":%d,\"sentry\":%d}", heap_caps_get_free_size(MALLOC_CAP_INTERNAL), ota.state, audioDB, asrText.c_str(), audioOn ? 1 : 0, sentryArmed ? 1 : 0); Serial1.println(b); }
    else if (c == "!SENTRY:ON") { sentryArmed = true; LOG_I("SENTRY ON"); Serial1.println("{\"sentry\":\"on\"}"); }
    else if (c == "!SENTRY:OFF") { sentryArmed = false; sentryTriggered = false; LOG_I("SENTRY OFF"); Serial1.println("{\"sentry\":\"off\"}"); }
    else if (c.startsWith("!SENTRY:THRESHOLD:")) {
        float t = atof(c.c_str() + 16);
        if (t >= 1.0f && t <= 10.0f) { sentryThreshold = t; LOG_I("SENTRY threshold=%.1f", t); }
    }
    else if (c.startsWith("!RADAR:")) {
        const char* resp = radar_cmd(&radarState, c.c_str());
        LOG_I("RADAR: %s", resp);
        Serial1.println(resp);
    }
}

bool capAudio() {
    if (!i2sOK) return false; size_t r = 0; int16_t tmp[PCM_N + 32];
    if (i2s_read(I2S_NUM_0, tmp, PCM_N * 2, &r, pdMS_TO_TICKS(10)) != ESP_OK) {
        static int i2sErr = 0;
        if (++i2sErr > 10) { i2sOK = false; LOG_E("I2S故障，关闭音频"); }
        return false;
    }
    int n = r / 2; int st = n > PCM_N ? n / PCM_N : 1; int w = 0;
    for (int i = 0; i < n && w < PCM_N; i += st) pcmBuf[w++] = tmp[i];
    while (w < PCM_N) pcmBuf[w++] = 0; return true;
}
uint8_t encADPCM(int16_t s) {
    int16_t d = s - adpcm.pred; uint8_t c = 0;
    if (d < 0) { c = 8; d = -d; } int16_t st = adpcmStep[adpcm.idx];
    int16_t dd = st >> 3; if (d >= st) { c |= 4; d -= st; dd = st >> 2; }
    if (d >= (st >> 1)) { c |= 2; d -= (st >> 1); } if (d >= dd) c |= 1;
    int16_t dt = st >> 3; if (c & 4) dt += st; if (c & 2) dt += st >> 1; if (c & 1) dt += st >> 2;
    if (c & 8) { adpcm.pred -= dt; if (adpcm.pred < -32768) adpcm.pred = -32768; }
    else { adpcm.pred += dt; if (adpcm.pred > 32767) adpcm.pred = 32767; }
    adpcm.idx += adpcmIdx[c & 7]; if (adpcm.idx < 0) adpcm.idx = 0; if (adpcm.idx > 88) adpcm.idx = 88;
    return c & 0x0F;
}
void encFrame(const int16_t* p, uint8_t* o, int n) { for (int i = 0; i < n; i += 2) *o++ = (encADPCM(p[i]) << 4) | encADPCM(p[i + 1]); }

void updateDisplay(int p, float t, float h, unsigned long uptime) {
    // 双页5秒切换
    unsigned long now = millis();
    if (now - dispPageTime > DISP_PAGE_MS) {
        dispPageTime = now;
        dispPage = (dispPage + 1) % 2;

    display.firstPage();
    do {
        if (dispPage == 0) drawMainPage(p, t, h, uptime);
        else drawDebugPage(p, t, h);
    } while (display.nextPage());
}

// ====== 哨兵独立任务(防gzip阻塞主循环) ======
void sentryTask(void* p) {
    while (1) {
        if (sentryTriggered) sentryProcess();
        vTaskDelay(pdMS_TO_TICKS(50));
    }

// ====== 哨兵模式 ======
void sentryInit() {
    if (!sentryRing) {
        // 从PSRAM分配环缓冲 (80KB)
        sentryRing = (uint8_t*)ps_malloc(SENTRY_RING_SIZE * SENTRY_ADPCM_FRAME);
        if (sentryRing) {
            memset(sentryRing, 0, SENTRY_RING_SIZE * SENTRY_ADPCM_FRAME);
            LOG_I("SENTRY: 环缓冲 %uKB 分配成功", 
                (SENTRY_RING_SIZE * SENTRY_ADPCM_FRAME) / 1024);
        } else {
            LOG_E("SENTRY: PSRAM分配失败! 哨兵不可用");
            return;
        }
    }
    sentryArmed = true;
    sentryTriggered = false;
    sentryWp = 0;
    LOG_I("SENTRY: 就绪 (阈值=%.1f×基线, TV感知=%s)", 
        sentryThreshold, sentryTVAware ? "开" : "关");
}

void sentryFeed(const uint8_t* adpcmFrame) {
    if (!sentryArmed || !sentryRing) return;
    // 写入环缓冲
    memcpy(sentryRing + sentryWp * SENTRY_ADPCM_FRAME, adpcmFrame, SENTRY_ADPCM_FRAME);
    sentryWp = (sentryWp + 1) % SENTRY_RING_SIZE;
    
    // 如果已触发，持续写入后触发区域
    if (sentryTriggered) {
        sentryPostCount++;
    }
}

static float lastSentryVol = 0;
static unsigned long lastSentryTrigger = 0;
void sentryTrigger(float peakVol) {
    if (sentryTriggered) return;
    if (millis() - lastSentryTrigger < 30000) return;
    float volDelta = fabs(peakVol - lastSentryVol);
    lastSentryVol = peakVol;
    if (volDelta < 5.0f && peakVol < 20.0f) return;  // 音量变化太小不触发
    lastSentryTrigger = millis();
    
    // 检查电视感知：电视开着 → 阈值自动提高
    float threshold = sentryThreshold;
    if (sentryTVAware) {
        // TODO: 通过红外接收头检测电视状态
        // bool tvOn = checkTVStatus();
        // if (tvOn) threshold *= 2.0f;
    }
    
    if (peakVol < audioBaseline * threshold && peakVol < 20.0f) return;
    
    // 🚨 触发哨兵
    sentryTriggered = true;
    sentryPostCount = 0;
    sentryStartMs = millis();
    LOG_I("SENTRY: 🚨 触发! vol=%.1f base=%.1f threshold=%.1f", 
        peakVol, audioBaseline, threshold);
    Serial1.printf("{\"sentry\":\"trigger\",\"vol\":%.1f,\"base\":%.1f}\n", 
        peakVol, audioBaseline);
}

void sentryProcess() {
    if (!sentryTriggered || !sentryRing) return;
    
    unsigned long elapsed = millis() - sentryStartMs;
    if (elapsed < (unsigned long)SENTRY_POST_MS) return;
    
    int totalPre = SENTRY_RING_SIZE;
    int totalPost = min(sentryPostCount, (int)(SENTRY_POST_MS / 20));
    int totalFrames = totalPre + totalPost;
    
    // 计算预触发区域起始位（环缓冲倒退10s）
    int preStart = (sentryWp - totalPre + SENTRY_RING_SIZE) % SENTRY_RING_SIZE;
    
    // 分配缓冲区：ADPCM数据 + gzip压缩输出
    size_t rawSize = totalFrames * SENTRY_ADPCM_FRAME;
    uint8_t* rawBuf = (uint8_t*)ps_malloc(rawSize);
    if (!rawBuf) {
        LOG_E("SENTRY: PSRAM分配失败,跳过上传");
        sentryTriggered = false; return;
    }
    
    // 1. 拷贝预触发数据（环缓冲读取）
    size_t off = 0;
    for (int i = 0; i < totalPre; i++) {
        int idx = (preStart + i) % SENTRY_RING_SIZE;
        memcpy(rawBuf + off, sentryRing + idx * SENTRY_ADPCM_FRAME, SENTRY_ADPCM_FRAME);
        off += SENTRY_ADPCM_FRAME;
    }
    // 2. 拷贝后触发数据（最新写入的帧）
    for (int i = 0; i < totalPost; i++) {
        int idx = (sentryWp - totalPost + i + SENTRY_RING_SIZE) % SENTRY_RING_SIZE;
        memcpy(rawBuf + off, sentryRing + idx * SENTRY_ADPCM_FRAME, SENTRY_ADPCM_FRAME);
        off += SENTRY_ADPCM_FRAME;
    }
    
    // 3. gzip压缩 (复用miniz.h)
    size_t compLen = rawSize + (rawSize >> 5) + 128;  // 压缩缓冲
    uint8_t* compBuf = (uint8_t*)ps_malloc(compLen);
    if (!compBuf) {
        free(rawBuf); sentryTriggered = false; return;
    }
    
    tdefl_compressor* comp = (tdefl_compressor*)malloc(sizeof(tdefl_compressor));
    if (comp) {
        tdefl_init(comp, NULL, NULL, TDEFL_DEFAULT_MAX_PROBES);
        size_t compActual = 0;
                esp_task_wdt_reset();  // gzip可能耗时,喂狗
        tdefl_status st = tdefl_compress(comp, rawBuf, &rawSize, compBuf, &compLen, TDEFL_FINISH);
        if (st == TDEFL_STATUS_DONE) {
            compActual = compLen;
            float ratio = 100.0f * compActual / rawSize;
            LOG_I("SENTRY: gzip %zu→%zu (%.0f%%)", rawSize, compActual, ratio);
            
            // 4. 通过Serial1上报事件信息（Air780转发到MQTT）
            // 格式: JSON事件描述 + base64音频数据（分块）
            Serial1.printf("{\"sentry\":\"event\",\"bytes\":%zu,\"comp\":%zu,\"vol\":%.1f}\n", 
                rawSize, compActual, audioVolume);
            // 后续：air780收到后请求下载完整数据
            // V1简化：只上报事件不传完整音频
        } else {
            LOG_E("SENTRY: gzip失败 st=%d", st);
        }
        free(comp);
    }
    
    free(compBuf);
    free(rawBuf);
    sentryTriggered = false;
    LOG_I("SENTRY: 事件完成, 哨兵复位");
}

void rptData(int p, float t, float h) {
    char b[320]; 
    // 取最近一个有效目标的距离和角度
    int d0 = 0, a0 = 0;
    for (int i = 0; i < 3; i++) {
        if (radarState.targets[i].active) {
            d0 = radarState.targets[i].distance;
            a0 = radarState.targets[i].angle;
            break;
        }
    }
    snprintf(b, sizeof(b), 
        "{\"p\":%d,\"t\":%.1f,\"h\":%d,\"rb\":%lu,\"hm\":%u,\"db\":%.0f,\"base\":%.1f,\"d\":%d,\"a\":%d}", 
        p, t, (int)h, radarBytes, esp_get_free_heap_size(), 
        audioDB, audioBaseline, d0, a0);
    // 通过UART发送（Air780 Lua代码会接收并转发MQTT）
    Serial1.println(b);
#ifdef USE_ML307C
    // ML307C模式：直接用AT指令发MQTT
    ml307_mqtt_publish("xiaobing/data", b);
#endif
}

void loop() {
    esp_task_wdt_reset();
    // GPIO心跳 - 每次loop翻转, Air780监控此引脚
    static int hb_toggle = 0;
    static unsigned long last_hb_toggle = 0;
    if (millis() - last_hb_toggle > 500) {  // 每秒2次
        last_hb_toggle = millis();
        hb_toggle = !hb_toggle;
        digitalWrite(HB_GPIO, hb_toggle);
    }
    unsigned long t = millis();
    if (uartOtaMode) { processUartOta(); return; }
    if (ota.state != OTA_IDLE) { processOTA(); return; }
    int pp = radar_parse(&radarState, PIN_RADAR_RX, PIN_RADAR_TX);
    pepCount = radarState.pepCount;
    rdDHT();
    static float volumeAvg = 0;
    static unsigned long lastAudioLog = 0;
    // 诊断模式但I2S故障时，每秒报一次状态
    if (audioDiagMode == 1 && !i2sOK && millis() - lastAudioLog > 1000) {
        lastAudioLog = millis();
        LOG_E("DIAG: I2S故障中，尝试重连...");
        setupI2S();
        Serial1.printf("{\"diag\":%d,\"i2s\":%d,\"msg\":\"I2S重试中...\"}\n", (millis()-audioDiagStart)/1000, i2sOK?1:0);
        if (!i2sOK && (millis()-audioDiagStart) > 10000) {
            audioDiagMode = 2;
            Serial1.println("{\"diag\":\"done\",\"verdict\":\"FAIL\",\"reason\":\"I2S初始化持续失败，检查硬件:WS=41 BCK=42 SD=2\"}");
            audioDiagMode = 0; audioOn = false;
        }
    }
    if (audioOn && i2sOK && capAudio()) {
        uint8_t aBuf[ADPCM_N + 2];
        aBuf[0] = (audio_seq >> 8) & 0xFF; aBuf[1] = audio_seq & 0xFF; audio_seq++;
        // 计算音量
        long long sum = 0;
        for (int i = 0; i < PCM_N; i++) sum += abs(pcmBuf[i]);
        float avgSample = (float)sum / PCM_N;  // 0~32767
        audioVolume = audioVolume * 0.9f + (avgSample / 100.0f) * 0.1f;
        
        // 分贝转换 dBFS = 20*log10(avg/32768), 范围-90~0
        float dbNow = 20.0f * log10f(avgSample / 32768.0f + 1e-10f);
        if (avgSample < 1) dbNow = -99.0f;  // 完全静音
        audioDB = audioDB * 0.7f + dbNow * 0.3f;
        
        // 每5秒打印音量到串口(方便观察麦克风是否工作)
    static unsigned long last_vol_print = 0;
    if (millis() - last_vol_print > 30000) {
        last_vol_print = millis();
        Serial.printf("[AUDIO] db=%.0f base=%.1f\n", audioDB, audioBaseline);
    }
        // 更新环境噪音基线（每10s取最低音量）
        if (millis() - audioBaselineTime > 10000) {
            audioBaselineTime = millis();
            // 更新环形缓冲区
            audioVolumeHistory[audioHistIdx] = audioVolume;
            audioHistIdx = (audioHistIdx + 1) % 10;
            // 基线 = 最低值×1.2（留余量）
            float minVol = audioVolumeHistory[0];
            for (int i = 1; i < 10; i++) if (audioVolumeHistory[i] < minVol) minVol = audioVolumeHistory[i];
            audioBaseline = minVol * 1.2f;
            if (audioBaseline < 0.5f) audioBaseline = 0.5f;  // 最低基线
            LOG_I("BASELINE: %.1f (min=%.1f)", audioBaseline, minVol);
        }
    
    // 诊断模式：每秒输出音量
        if (audioDiagMode == 1 && millis() - lastAudioLog > 1000) {
            lastAudioLog = millis();
            audioDiagCount++;
            int elapsed = (millis() - audioDiagStart) / 1000;
            audioDiagSum += audioVolume;
            if (audioVolume < audioDiagMin) audioDiagMin = audioVolume;
            if (audioVolume > audioDiagMax) audioDiagMax = audioVolume;
            float avg = audioDiagSum / audioDiagCount;
            LOG_I("DIAG[%d/%d]: vol=%.1f min=%.1f max=%.1f avg=%.1f", elapsed, 30, audioVolume, audioDiagMin, audioDiagMax, avg);
            Serial1.printf("{\"diag\":%d,\"vol\":%.1f,\"min\":%.1f,\"max\":%.1f,\"avg\":%.1f,\"i2s\":%d}\n", elapsed, audioVolume, audioDiagMin, audioDiagMax, avg, i2sOK ? 1 : 0);
            if (audioDiagCount >= 30) {
                audioDiagMode = 2;
                const char* verdict;
                if (avg > 5.0f) verdict = "OK(正常)";
                else if (avg > 0.5f) verdict = "OK(安静)";
                else if (avg > 0.1f) verdict = "WARNING(弱信号)";
                else verdict = "FAIL(无声，检查硬件)";
                LOG_I("DIAG DONE: %s (avg=%.1f)", verdict, avg);
                Serial1.printf("{\"diag\":\"done\",\"verdict\":\"%s\",\"avg\":%.1f,\"min\":%.1f,\"max\":%.1f}\n", verdict, avg, audioDiagMin, audioDiagMax);
                audioDiagMode = 0;
            }
        }
        
        encFrame(pcmBuf, aBuf + 2, PCM_N);
        
        // 哨兵：持续Feed环缓冲 + 检测触发
        sentryFeed(aBuf + 2);
        sentryTrigger(audioVolume);
        
        // 每10帧发一次音频到MQTT（base64，Air780兼容）
    static int audioSendCnt = 0;
    if (++audioSendCnt >= 10) {
        audioSendCnt = 0;
        int len = ADPCM_N;
        size_t outLen = ((len + 2) / 3) * 4 + 1;
        uint8_t b64[outLen];
        size_t olen = 0;
        mbedtls_base64_encode(b64, outLen, &olen, aBuf + 2, len);
        b64[olen] = 0;
        Serial1.println("{\"a\":\"" + String((char*)b64) + "\",\"s\":" + String(audio_seq) + "}");
    }
    }
    // IR学习检测
    if (irrecv.decode(&irResult)) {
        String code = String(irResult.value, HEX);
        LOG_I("IR收到: %s", code.c_str());
        Serial1.println("{\"ir\":\"" + code + "\"}");
        irrecv.resume();
    }
    // 从USB串口(Serial)接收命令
    if (Serial.available()) {
        static String serialBuf;
        char c = Serial.read();
        if (c == '\n' || c == '\r') {
            if (serialBuf.length() > 0) { procCmd(serialBuf); serialBuf = ""; }
        } else if (serialBuf.length() < 256) {
            serialBuf += c;
        }
    }
    if (t % 100 == 0) { updateDisplay(pp, lt, lh, millis()); parseUART(); }
    // 数据汇总：每60秒上报一次聚合值
    if (t % (60 * 1000) == 0) { rptData(pp, lt, lh); monSystem(); ml307_publish_health(); }
#ifdef USE_ML307C
    ml307_loop();  // ML307C自动重连
    if (t % 5000 == 0) ml307_debug_dump();  // 5秒印一次ML307C状态，减少刷屏
#endif
    vTaskDelay(pdMS_TO_TICKS(1));
}

// 🦞 虾兵 · LD2450雷达驱动 v2.0 实现
#include "ld2450_driver.h"

// ====== Signed Magnitude 解码（ESPHome标准） ======
// LD2450数据格式：高位字节bit15=符号位(1=负数), bits14-0=绝对值
static int16_t decodeCoord(uint8_t lo, uint8_t hi) {
    int16_t mag = (hi & 0x7F) << 8 | lo;
    if ((hi & 0x80) == 0) mag = -mag;  // 符号位=0 → 取反
    return mag;  // 单位: mm
}

// ====== 初始化 ======
void radar_init() {
    // 默认配置，由调用者在setup中按需修改
}

// ====== 坐标旋转 ======
void radar_rotate_coords(int16_t* x, int16_t* y, float deg) {
    if (fabs(deg) < 0.5f) return;  // 角度太小时跳过
    float rad = deg * M_PI / 180.0f;
    float cos_a = cosf(rad), sin_a = sinf(rad);
    int16_t nx = (int16_t)((*x) * cos_a - (*y) * sin_a);
    int16_t ny = (int16_t)((*x) * sin_a + (*y) * cos_a);
    *x = nx; *y = ny;
}

// ====== 点是否在区域内 ======
bool radar_inside_zone(RadarState* state, int16_t x, int16_t y) {
    if (!state->zone_enabled) return true;  // 没开区域过滤 = 全区域
    if (state->zone_count == 0) return true;
    for (int i = 0; i < state->zone_count; i++) {
        Zone* z = &state->zones[i];
        if (!z->enabled) continue;
        if (x >= z->x1 && x <= z->x2 && y >= z->y1 && y <= z->y2)
            return true;
    }
    return false;  // 不在任何区域
}

// ====== 设置区域 ======
void radar_set_zone(RadarState* state, uint8_t idx, int16_t x1, int16_t y1, int16_t x2, int16_t y2) {
    if (idx >= 9) return;
    state->zones[idx] = {x1, y1, x2, y2, true};
    if (idx >= state->zone_count) state->zone_count = idx + 1;
    state->zone_enabled = true;
}

void radar_clear_zones(RadarState* state) {
    for (int i = 0; i < 9; i++) state->zones[i] = {0,0,0,0,false};
    state->zone_count = 0;
    state->zone_enabled = false;
}

void radar_set_rotation(RadarState* state, float deg) { state->rotation_deg = deg; }
void radar_set_timeout(RadarState* state, uint32_t ms) { state->presence_timeout = ms; }

// ====== 核心：解析一帧雷达数据 ======
// 返回: >=0 = 当前人数, -1 = 帧头错误, -2 = 帧尾错误, -3 = 缓冲区不够
int radar_parse(RadarState* state, int rx_pin, int tx_pin) {
    (void)tx_pin; // 只有RX
    unsigned long now = millis();
    static uint8_t buf[LD2450_FRAME_LEN];
    static int idx = 0;
    bool gotFrame = false;
    
    // 持续读串口直到没数据
    while (Serial2.available()) {
        uint8_t b = Serial2.read();
        
        // 帧同步: AA FF 03 00
        if (idx == 0 && b != LD2450_HEADER_1) continue;
        if (idx == 1 && b != LD2450_HEADER_2) { idx = 0; continue; }
        if (idx == 2 && b != LD2450_HEADER_3) { idx = 0; continue; }
        if (idx == 3 && b != LD2450_HEADER_4) { idx = 0; continue; }
        
        buf[idx++] = b;
        if (idx < LD2450_FRAME_LEN) continue;
        idx = 0;
        
        // 校验帧尾: 55 CC
        if (buf[28] != LD2450_FOOTER_1 || buf[29] != LD2450_FOOTER_2)
            continue;
        
        gotFrame = true;
        state->totalFrames++;
        
        // ---- 解析3个目标 ----
        for (int i = 0; i < LD2450_MAX_TARGETS; i++) {
            int o = 4 + i * 8;
            int16_t rx = decodeCoord(buf[o],   buf[o+1]);
            int16_t ry = decodeCoord(buf[o+2], buf[o+3]);
            int16_t rs = decodeCoord(buf[o+4], buf[o+5]);
            uint16_t res = (buf[o+7] << 8) | buf[o+6];
            
            // 坐标旋转
            radar_rotate_coords(&rx, &ry, state->rotation_deg);
            
            // 判断是否在检测区域内
            bool inZone = radar_inside_zone(state, rx, ry);
            
            // 目标有效性判断: dist>0 + resolution>0 + 在区域内
            int16_t adx = abs(rx), ady = abs(ry);
            bool valid = (adx + ady > 0) && (res > 0) && inZone;
            
            state->targets[i].x = rx;
            state->targets[i].y = ry;
            state->targets[i].speed = rs;
            state->targets[i].resolution = res;
            state->targets[i].distance = (uint16_t)roundf(hypotf(rx, ry));
            state->targets[i].angle = (rx == 0 && ry == 0) ? 0 :
                (int16_t)roundf(atan2f(rx, ry) * 180.0f / M_PI);
            state->targets[i].moving = (rs != 0);
            
            // 稳定性计数器：连续有效才确认（避免单帧抖动）
            if (valid) {
                if (state->stabilityCount[i] < 255)
                    state->stabilityCount[i]++;
                if (state->stabilityCount[i] >= TARGET_STABILITY_FRAMES) {
                    state->lastSeen[i] = now;
                    state->targets[i].active = true;
                } else {
                    state->targets[i].active = (now - state->lastSeen[i] < state->presence_timeout);
                }
            } else {
                if (state->stabilityCount[i] > 0)
                    state->stabilityCount[i]--;
                state->targets[i].active = (now - state->lastSeen[i] < state->presence_timeout);
            }
        }
    }
    
    // 计算人数：在超时时间内且active=true
    state->pepCount = 0;
    for (int i = 0; i < LD2450_MAX_TARGETS; i++) {
        if (state->targets[i].active && 
            (now - state->lastSeen[i] < state->presence_timeout)) {
            state->pepCount++;
        }
    }
    
    return state->pepCount;
}

// ====== 配置命令 ======
// 格式:
//   !RADAR:TIMEOUT:5000          → 设置存在超时(ms)
//   !RADAR:ROTATE:15             → 设置旋转角度(度)
//   !RADAR:ZONE:1:0,0,2000,3000  → 设置区域1(x1,y1,x2,y2)
//   !RADAR:ZONE:CLEAR            → 清空所有区域
//   !RADAR:STATUS                → 打印当前配置
const char* radar_cmd(RadarState* state, const char* cmd) {
    static char response[128];
    
    if (strncmp(cmd, "!RADAR:TIMEOUT:", 15) == 0) {
        uint32_t ms = atoi(cmd + 15);
        if (ms >= 1000 && ms <= 60000) {
            state->presence_timeout = ms;
            snprintf(response, sizeof(response), "RADAR_TIMEOUT=%u", ms);
            return response;
        }
        return "RADAR_ERR:timeout 1000-60000";
    }
    
    if (strncmp(cmd, "!RADAR:ROTATE:", 14) == 0) {
        float deg = atof(cmd + 14);
        if (deg >= -180.0f && deg <= 180.0f) {
            state->rotation_deg = deg;
            snprintf(response, sizeof(response), "RADAR_ROTATE=%.1f", deg);
            return response;
        }
        return "RADAR_ERR:deg -180~180";
    }
    
    if (strncmp(cmd, "!RADAR:ZONE:", 12) == 0) {
        const char* p = cmd + 12;
        if (strcmp(p, "CLEAR") == 0) {
            radar_clear_zones(state);
            return "RADAR_ZONE=CLEAR";
        }
        // !RADAR:ZONE:IDX:X1,Y1,X2,Y2
        int idx, x1, y1, x2, y2;
        if (sscanf(p, "%d:%d,%d,%d,%d", &idx, &x1, &y1, &x2, &y2) == 5) {
            if (idx >= 0 && idx < 9) {
                radar_set_zone(state, idx, x1, y1, x2, y2);
                snprintf(response, sizeof(response), "RADAR_ZONE%d=%d,%d,%d,%d", idx, x1, y1, x2, y2);
                return response;
            }
        }
        return "RADAR_ERR:zone format IDX:X1,Y1,X2,Y2";
    }
    
    if (strcmp(cmd, "!RADAR:STATUS") == 0) {
        snprintf(response, sizeof(response), 
            "RADAR: people=%d timeout=%u rot=%.1f zones=%d frames=%lu",
            state->pepCount, state->presence_timeout, state->rotation_deg,
            state->zone_count, state->totalFrames);
        return response;
    }
    
    return "RADAR_ERR:unknown cmd";
}

#include <esp_task_wdt.h>
static bool ml307_at_cmd(const char* cmd, const char* expect, unsigned long timeout_ms) {
    while (Serial1.available()) Serial1.read();
    Serial1.printf("%s\r\n", cmd);
    unsigned long start = millis();
    char buf[AT_BUF_SIZE] = {0};
    int pos = 0;
    while (millis() - start < timeout_ms) {
        esp_task_wdt_reset();  // 喂狗！AT等待可能很久
        while (Serial1.available() && pos < AT_BUF_SIZE - 1) {
            char c = Serial1.read();
            if (c == '\r' || c == '\n') continue;
            buf[pos++] = c;
            buf[pos] = 0;
            if (strstr(buf, expect)) return true;
        }
    }
    LOG_ML307("AT超时: cmd=%s expect=%s buf=[%s]", cmd, expect, buf);
    return false;
}

static void ml307_passthru() {
    LOG_ML307("=== UART透传模式 ===");
    LOG_ML307("串口助手直接与ML307C通信，输入 EXIT 退出");
    unsigned long start = millis();
    while (millis() - start < 60000) {  // 最多透传60秒
        esp_task_wdt_reset();  // 喂狗！防止看门狗10秒复位
        if (Serial.available()) {
            char c = Serial.read();
            if (c == 0x03) break;  // Ctrl+C 退出
            Serial1.write(c);
        }
        if (Serial1.available()) {
            Serial.write(Serial1.read());
        }
        // 检测 EXIT 命令
        static String passthruBuf;
        if (Serial.available()) {
            char c = Serial.read();
            Serial1.write(c);
            passthruBuf += c;
            if (passthruBuf.endsWith("EXIT\r\n") || passthruBuf.endsWith("EXIT\n")) break;
        }
    }
    LOG_ML307("透传模式退出");
}

// ---------- MQTT ----------
static bool ml307_mqtt_connect() {
    LOG_ML307("MQTT连接 %s:%d...", ML307_MQTT_HOST, ML307_MQTT_PORT);
    // 先强制断开可能残留的连接
    ml307_at_cmd("AT+MQTTDISC=0", "OK", 2000);
    delay(200);
    // 配置: clean=1清会话 + keepalive=120 + pingresp
    ml307_at_cmd("AT+MQTTCFG=\"clean\",0,1", "OK", 3000);
    ml307_at_cmd("AT+MQTTCFG=\"keepalive\",0,120", "OK", 3000);
    ml307_at_cmd("AT+MQTTCFG=\"pingresp\",0,1", "OK", 3000);
    for (int i = 0; i < 3; i++) {
        char client_id[32];
        snprintf(client_id, sizeof(client_id), "xiabing_%d", i + 1);
        char cmd[160];
        snprintf(cmd, sizeof(cmd),
            "AT+MQTTCONN=0,\"%s\",%d,\"%s\"",  // 去空user/pass,防607
            ML307_MQTT_HOST, ML307_MQTT_PORT, client_id);
        // 检查OK或+MQTTURC:"conn",0,0(模块可能先报ERROR再发URC确认)
        // 同时等待两种成功回应
        while (Serial1.available()) Serial1.read();
        Serial1.printf("%s\r\n", cmd);
        unsigned long start = millis();
        char buf[AT_BUF_SIZE] = {0};
        int pos = 0;
        bool mqtt_ok = false;
        while (millis() - start < 20000) {
            esp_task_wdt_reset();
            while (Serial1.available() && pos < AT_BUF_SIZE - 1) {
                char cc = Serial1.read();
                if (cc == '\r' || cc == '\n') continue;
                buf[pos++] = cc;
                buf[pos] = 0;
                if (strstr(buf, "OK") || strstr(buf, "\"conn\",0,0")) { mqtt_ok = true; break; }
            }
            if (mqtt_ok) break;
        }
        if (mqtt_ok) {
            ml307_mqtt_ok = true;
            LOG_ML307("MQTT连接成功！");
            // 等CONN URC到达(模块异步发送,需等待)
            delay(500);
            while (Serial1.available()) Serial1.read();
            // 订阅命令通道
            Serial1.printf("AT+MQTTSUB=0,\"xiaobing/cmd\",1\r\n");
            unsigned long _sub_start = millis();
            bool _sub_ok = false;
            char _sub_buf[128] = {0};
            int _sub_p = 0;
            while (millis() - _sub_start < 5000) {
                if (Serial1.available()) {
                    char cc = Serial1.read();
                    if (cc != '\\r' && cc != '\\n' && _sub_p < 127) _sub_buf[_sub_p++] = cc;
                    _sub_buf[_sub_p] = 0;
                    if (strstr(_sub_buf, "OK") || strstr(_sub_buf, "suback")) { _sub_ok = true; break; }
                }
            }
            LOG_ML307("订阅%s", _sub_ok ? "成功" : "失败");
            return true;
        }
        LOG_ML307("MQTT未成功 buf=[%s]", buf);
        LOG_ML307("MQTT连接失败(尝试%d/3)...", i + 1);
        delay(3000);
    }
    ml307_mqtt_ok = false;
    LOG_ML307("MQTT连接失败！检查broker(%s:%d)是否在线", ML307_MQTT_HOST, ML307_MQTT_PORT);
    return false;
}

static bool ml307_mqtt_publish(const char* topic, const char* data) {
    if (!ml307_mqtt_ok) return false;
    // 先delay 200ms确保连接已稳定
    delay(200);
    // 清空URC消息
    while (Serial1.available()) Serial1.read();
    // 内联格式: AT+MQTTPUB=0,<topic>,0,0,0,<len>,<data>
    int len = strlen(data);
    char cmd[600];
    snprintf(cmd, sizeof(cmd),
        "AT+MQTTPUB=0,\"%s\",0,0,0,%d,\"%s\"",
        topic, len, data);
    Serial1.printf("%s\r\n", cmd);
    unsigned long start = millis();
    char rbuf[256] = {0};
    int rp = 0;
    while (millis() - start < 10000) {
        esp_task_wdt_reset();
        if (Serial1.available()) {
            char cc = Serial1.read();
            if (cc != '\\r' && cc != '\\n' && rp < 255) rbuf[rp++] = cc;
            rbuf[rp] = 0;
            if (strstr(rbuf, "OK")) { ml307_tx_bytes += len; return true; }
            if (strstr(rbuf, "ERROR")) break;
        }
    }
    LOG_ML307("MQTT发布失败 topic=%s rc=[%s]", topic, rbuf);
    return false;
}

// ---------- ML307C 流式OTA ----------
static void ml307_ota() {
    LOG_ML307("流式OTA开始...");
    // 切921600加速（注意: IPR命令后模块立即切波特率，不等OK）
    LOG_ML307("切921600波特率...");

    // 验证握手
    for (int i = 0; i < 3; i++) {
        if (ml307_at_cmd("AT", "OK", 2000)) { LOG_ML307("921600握手OK"); break; }
        LOG_ML307("握手重试 %d/3", i+1);
    }
    // 获取OTA分区
    const esp_partition_t* part = esp_ota_get_next_update_partition(NULL);
    if (!part) { LOG_ML307("无OTA分区"); return; }
    LOG_ML307("写 %s @0x%x", part->label, part->address);
    
    // 打开TCP到下载服务器
    char cmd[128];
    snprintf(cmd, sizeof(cmd), "AT+MIPOPEN=1,\"TCP\",\"%s\",3001,120,0", ML307_MQTT_HOST);
    if (!ml307_at_cmd(cmd, "OK", 15000)) { LOG_ML307("TCP连接失败"); return; }
    LOG_ML307("TCP已连接，请求固件...");
    
    // 用MIPSEND发HTTP GET
    // 清URC残留(+MIPOPEN URC会混进来)
    while (Serial1.available()) Serial1.read();
    delay(200);
    // 发HTTP GET（等>提示符后发裸数据）
    if (!ml307_at_cmd("AT+MIPSEND=1,68", ">", 10000)) {
        // 可能直接收到OK(模块自动发完),检查是否有OK
        ml307_at_cmd("AT+MIPSEND=1,68", "OK", 5000);
        LOG_ML307("MIPSEND继续...");
    }
    Serial1.printf("GET /firmware/merged-v0307.bin HTTP/1.0\\r\\nHost: %s\\r\\n\\r\\n", ML307_MQTT_HOST);
    delay(200);
    // 等发送完成
    ml307_at_cmd("", "OK", 5000);
    
    // 解析HTTP头获取Content-Length
    unsigned long total_size = 0;
    unsigned long start = millis();
    size_t written = 0, last_prog = 0;
    bool in_body = false;
    uint8_t buf[1024];
    int bp = 0;
    char header_buf[512];
    int hp = 0;
    
    // 先收HTTP头(最多5秒)
    unsigned long head_timeout = millis() + 5000;
    while (millis() < head_timeout) {
        while (Serial1.available()) {
            char cc = Serial1.read();
            if (hp < (int)sizeof(header_buf)-1) header_buf[hp++] = cc;
            header_buf[hp] = 0;
            // 检测头结束
            if (hp >= 4 && header_buf[hp-4]=='\\r' && header_buf[hp-3]=='\\n' && 
                header_buf[hp-2]=='\\r' && header_buf[hp-1]=='\\n') {
                in_body = true;
                // 提取Content-Length
                char* cl = strstr(header_buf, "Content-Length:");
                if (cl) total_size = atol(cl + 15);
                break;
            }
        }
        if (in_body) break;
    }
    
    if (!in_body) { LOG_ML307("HTTP头解析失败"); ml307_at_cmd("AT+MIPCLOSE=1","OK",3000); return; }
    
    // 估算时间
    unsigned long baud = 921600;  // 已切到921600
    int est_sec = (total_size > 0) ? (total_size * 10 / baud) + 5 : 60;
    LOG_ML307("固件 %uKB @%lubps ~%d秒", total_size/1024, baud, est_sec);
    
    // 开始流式OTA
    esp_ota_handle_t oh;
    if (esp_ota_begin(part, OTA_SIZE_UNKNOWN, &oh) != ESP_OK) { LOG_ML307("OTA begin失败"); return; }
    
    // 先把头剩下的数据(如果有)写进去
    if (hp > 4 && in_body) {
        // header已包含结尾\\r\\n\\r\\n, body从hp位置开始
        // 但hp指向头结束位置, 实际body数据可能在buf里... 简化: 重新读
    }
    
    unsigned long ota_start = millis();
    while (millis() - start < 180000) {
        esp_task_wdt_reset();
        if (Serial1.available()) {
            buf[bp++] = Serial1.read();
            if (bp >= 256) {
                esp_ota_write(oh, buf, bp);
                written += bp; bp = 0;
                if (written - last_prog > 20000) {
                    last_prog = written;
                    int pct = (total_size > 0) ? written * 100 / total_size : 0;
                    int elapsed = (millis() - ota_start) / 1000;
                    int eta = (pct > 0) ? elapsed * (100 - pct) / pct : est_sec;
                    LOG_ML307("OTA %d%% %uKB/%uKB %ds ETA~%ds", pct, written/1024, total_size/1024, elapsed, eta);
                }
            }
        }
    }
    if (bp > 0) esp_ota_write(oh, buf, bp);
    written += bp;
    
    esp_ota_end(oh);
    esp_ota_set_boot_partition(part);
    ml307_at_cmd("AT+MIPCLOSE=1","OK",3000);
    LOG_ML307("OTA完成 %uKB/%uKB %d秒 重启...", written/1024, total_size/1024, (millis()-ota_start)/1000);
    delay(500);
    ESP.restart();
}

// ---------- 重连+异步初始化 ----------
// 在 loop() 中定期调用，负责初始化和掉线重连
static void ml307_loop() {
    esp_task_wdt_reset();
    
    // 首次初始化（异步状态机，不阻塞setup）
    if (!ml307_init_done) {
        switch (ml307_init_step) {
            case 0: // 等模块启动
                LOG_ML307("等待模块启动...");
                ml307_step_start = millis();
                ml307_init_step = 1;
                break;
            case 1: // 5秒延时
                if (millis() - ml307_step_start < 5000) break;
                LOG_ML307("检测模块...");
                ml307_step_start = millis();
                ml307_init_step = 2;
                break;
            case 2: // 扫波特率（ML307C默认可能不是115200）
                if (ml307_baud_trial < ML307_BAUD_COUNT) {
                    unsigned long br = ML307_BAUD_TRIAL[ml307_baud_trial];
                    LOG_ML307("尝试波特率 %lu...", br);
                    Serial1.flush();
                    Serial1.updateBaudRate(br);
                    delay(100);
                    if (ml307_at_cmd("AT", "OK", 2000)) {
                        LOG_ML307("模块响应OK! 波特率=%lu", br);
                        ml307_step_start = millis();
                        ml307_init_step = 3;
                    } else {
                        ml307_baud_trial++;
                    }
                } else {
                    LOG_ML307("所有波特率均无响应，跳过4G");
                    LOG_ML307("检查: ①VIN供电5V/≥2A ②EN与VIN短接 ③BT脚不接地");
                    ml307_init_done = true;
                    ml307_init_success = false;
                }
                break;
            case 3: // SIM卡
                if (ml307_at_cmd("AT+CPIN?", "READY", 5000)) {
                    LOG_ML307("SIM卡就绪");
                    ml307_step_start = millis();
                    ml307_init_step = 4;
                } else {
                    LOG_ML307("SIM卡未就绪，跳过4G");
                    ml307_init_done = true;
                    ml307_init_success = false;
                }
                break;
            case 4: // CSQ
                ml307_at_cmd("AT+CSQ", "+CSQ:", 3000);
                ml307_init_step = 5;
                break;
            case 5: // APN — 自动轮询
            {
                if (ml307_apn_trial < ML307_APN_COUNT) {
                    char apnCmd[64];
                    snprintf(apnCmd, sizeof(apnCmd), "AT+CGDCONT=1,\"IPV4V6\",\"%s\"", ML307_APN_LIST[ml307_apn_trial]);
                    if (ml307_at_cmd(apnCmd, "OK", 5000)) {
                        LOG_ML307("APN OK: [%s]", ML307_APN_LIST[ml307_apn_trial]);
                        ml307_apn_trial = 0;  // 重置
                        ml307_init_step = 6;
                    } else {
                        ml307_apn_trial++;
                    }
                } else {
                    ml307_apn_trial = 0;
                    ml307_init_step = 6;  // 都不行也继续，MIPCALL可能自动分配
                }
                break;
            }
            case 6: // CREG
                ml307_at_cmd("AT+CREG?", "+CREG:", 3000);
                ml307_init_step = 7;
                break;
            case 7: // CGATT
                if (ml307_at_cmd("AT+CGATT=1", "OK", 15000)) {
                    ml307_network_ok = true;
                    LOG_ML307("网络注册成功！");
                    ml307_step_start = millis();
                    ml307_init_step = 8;
                } else {
                    LOG_ML307("网络注册失败，跳过4G");
                    ml307_init_done = true;
                    ml307_init_success = false;
                }
                break;
            case 8: { // 激活数据通道（通信流程示例完整流程）
                LOG_ML307("激活数据通道...");
                ml307_at_cmd("AT+CSQ", "+CSQ:", 3000);
                // 通信流程示例: 
                // 1. AT+CFUN? 确认协议栈已开启
                // 2. AT+CGDCONT=1,"IPV4V6","CMNET" → 配置PDP上下文
                // 3. AT+MIPCALL? → 检查是否已自动拨号
                // 4. AT+MIPCALL=1,1 → 激活数据连接
                ml307_at_cmd("AT+CFUN?", "1", 3000);
                
                // 自动轮询APN: 空→CMNET→3GNET→CTNET
                bool pdp_ok = false;
                for (int a = 0; a < ML307_APN_COUNT; a++) {
                    char pdp_cmd[64];
                    snprintf(pdp_cmd, sizeof(pdp_cmd), "AT+CGDCONT=1,\"IPV4V6\",\"%s\"", ML307_APN_LIST[a]);
                    if (ml307_at_cmd(pdp_cmd, "OK", 3000)) {
                        LOG_ML307("PDP APN: [%s]", ML307_APN_LIST[a]);
                        pdp_ok = true;
                        break;
                    }
                }
                LOG_ML307("PDP配置%s", pdp_ok ? "完成" : "失败(继续试)");
                
                // 检查是否已自动拨号
                if (ml307_at_cmd("AT+MIPCALL?", "+MIPCALL: 1,", 3000)) {
                    LOG_ML307("数据通道已激活(自动拨号)");
                    ml307_init_step = 9;
                } else if (ml307_at_cmd("AT+MIPCALL=1,1", "OK", 30000)) {
                    LOG_ML307("数据通道激活成功");
                    ml307_init_step = 9;
                } else {
                    LOG_ML307("MIPCALL失败，尝试CGACT...");
                    if (ml307_at_cmd("AT+CGACT=1,1", "OK", 10000)) {
                        LOG_ML307("数据通道激活成功(CGACT)");
                    } else {
                        LOG_ML307("数据通道激活失败，检查SIM卡流量");
                        LOG_ML307("可进透传(!ML307)手动配APN: AT+CGDCONT=1,\"IPV4V6\",\"你的APN\"");
                    }
                    ml307_init_step = 9;
                }
                break;
            } // case 8 end
            case 9: // MQTT
                ml307_init_done = true;
                ml307_init_success = true;
                LOG_ML307("4G初始化完成，连接MQTT...");
                ml307_mqtt_connect();
                break;
        }
        return;
    }
    
    // 初始化完成后的重连逻辑
    if (ml307_is_connected()) return;
    if (millis() - ml307_last_retry < 30000) return;
    ml307_last_retry = millis();
    
    LOG_ML307("检测到离线，尝试重连...");
    
    if (!ml307_network_ok) {
        // 检查网络是否还在
        int net = ml307_check_network();
        if (net == 1) {
            ml307_network_ok = true;
            LOG_ML307("网络已恢复");
        } else if (net == 0) {
            // 尝试重新注册
            LOG_ML307("网络断开，重新注册...");
            for (int i = 0; i < 3; i++) {
                if (ml307_at_cmd("AT+CGATT=1", "OK", 15000)) {
                    ml307_network_ok = true;
                    LOG_ML307("网络重注册成功");
                    break;
                }
            }
        }
        // net==-1（无响应）不做处理，可能模块挂了
    }
    
    if (ml307_network_ok && !ml307_mqtt_ok) {
        ml307_mqtt_connect();
    }
}

// 🦞 虾兵 · 多传感器融合引擎
// 基于10秒窗口 + 5分钟汇总的多模态置信度判断
// 雷达 × CSI × 音频 × BLE

#pragma once

#include <Arduino.h>

// 融合参数
#define FUSION_WINDOW_SEC   10    // 采样窗口（秒）
#define FUSION_CYCLE_SEC    300   // 汇总周期（5分钟）
#define FUSION_CONF_HIGH    70    // 有人阈值(%)
#define FUSION_CONF_LOW     30    // 无人阈值(%)
#define FUSION_HOLD_MS      600000 // 无人持续10分钟后关空调

// 各传感器权重
// 传感器权重（CSI/BLE硬件当前不可用，权重重新分配给雷达和音频）
#define WEIGHT_RADAR  0.80f
#define WEIGHT_CSI    0.00f
#define WEIGHT_AUDIO  0.20f
#define WEIGHT_BLE    0.00f

struct FusionState {
    // 10秒窗口采样
    int radar_samples = 0;      // 采样次数
    int radar_present = 0;      // 检测到有人次数
    int csi_samples = 0;
    int csi_motion = 0;
    int audio_samples = 0;      // 音频有活跃的采样次数（计数模式）
    bool audio_active = false;  // 有声音（计算后结果）
    bool ble_present = false;   // 有蓝牙设备
    
    // 计算结果
    float confidence = 0;       // 当前置信度 0-100
    bool occupied = false;      // 综合判断有人
    
    // 5分钟汇总
    float avg_confidence = 0;   // 周期平均置信度
    int cycle_count = 0;
    
    // 状态
    unsigned long window_start = 0;
    unsigned long last_occupied_time = 0;
    unsigned long last_report = 0;
};

class FusionEngine {
public:
    FusionEngine() {}
    
    void begin() {
        state_.window_start = millis();
        state_.last_report = millis();
    }
    
    // 雷达上报（每秒调用）
    void report_radar(int count) {
        state_.radar_samples++;
        if (count > 0) state_.radar_present++;
    }
    
    // CSI上报
    void report_csi(float level, bool motion) {
        state_.csi_samples++;
        if (motion || level > 0.5) state_.csi_motion++;
    }
    
    // 音频上报
    // 音频上报（改为计数模式，非粘性标志 - 每个窗口内重置）
    void report_audio(bool active) {
        if (active) state_.audio_samples++;
    }
    
    // BLE上报
    void report_ble(bool present) {
        state_.ble_present = present;
    }
    
    // 每100ms在loop调用
    void process() {
        unsigned long now = millis();
        unsigned long elapsed = (now - state_.window_start) / 1000;
        
        if (elapsed >= FUSION_WINDOW_SEC) {
            // 计算当前窗口置信度
            calculate_confidence();
            
            // 重置窗口
            state_.window_start = now;
            state_.radar_samples = 0;
            state_.radar_present = 0;
            state_.csi_samples = 0;
            state_.csi_motion = 0;
            state_.audio_samples = 0;
            state_.audio_active = false;
            
            // 5分钟汇总
            state_.cycle_count++;
            state_.avg_confidence += state_.confidence;
            
            if (now - state_.last_report >= FUSION_CYCLE_SEC * 1000) {
                state_.avg_confidence /= state_.cycle_count;
                state_.cycle_count = 0;
                report_summary();
                state_.avg_confidence = 0;
                state_.last_report = now;
            }
        }
    }
    
    float get_confidence() { return state_.confidence; }
    bool is_occupied() { return state_.occupied; }
    
    // 获取JSON上报字符串
    String get_json() {
        char buf[256];
        snprintf(buf, sizeof(buf),
            "{\"fusion\":{\"conf\":%.0f,\"occ\":%d,\"r\":%d/%d,\"c\":%d/%d,\"a\":%d,\"b\":%d}}",
            state_.confidence, state_.occupied ? 1 : 0,
            state_.radar_present, state_.radar_samples,
            state_.csi_motion, state_.csi_samples,
            state_.audio_active ? 1 : 0,
            state_.ble_present ? 1 : 0
        );
        return String(buf);
    }

private:
    FusionState state_;
    
    void calculate_confidence() {
        float radar_conf = 0, csi_conf = 0, audio_conf = 0, ble_conf = 0;
        
        // 雷达置信度
        if (state_.radar_samples > 0)
            radar_conf = (float)state_.radar_present / state_.radar_samples * 100;
        
        // CSI置信度
        if (state_.csi_samples > 0)
            csi_conf = (float)state_.csi_motion / state_.csi_samples * 100;
        
        // 音频置信度（按采样帧中活跃比例计算）
        if (state_.radar_samples > 0)
            audio_conf = (float)state_.audio_samples / state_.radar_samples * 50;
        else
            audio_conf = state_.audio_samples > 0 ? 50 : 0;
        
        // BLE置信度
        ble_conf = state_.ble_present ? 70 : 0;
        
        // 综合
        state_.confidence = radar_conf * WEIGHT_RADAR + csi_conf * WEIGHT_CSI 
                          + audio_conf * WEIGHT_AUDIO + ble_conf * WEIGHT_BLE;
        
        unsigned long now = millis();
        
        // 有人/无人判断
        if (state_.confidence > FUSION_CONF_HIGH) {
            if (!state_.occupied) {
                state_.occupied = true;
                state_.last_occupied_time = now;
            }
        } else if (state_.confidence < FUSION_CONF_LOW) {
            if (state_.occupied && (now - state_.last_occupied_time > FUSION_HOLD_MS)) {
                state_.occupied = false;
            }
        }
    }
    
    void report_summary() {
        // 5分钟汇总，可通过Serial1上报
        Serial1.printf("{\"fusion_summary\":{\"avg_conf\":%.0f,\"cycle\":%ds}}\n",
            state_.avg_confidence, FUSION_CYCLE_SEC);
    }
};

extern FusionEngine fusion_engine;

// 全局SPI
static SPIClass tft_spi(HSPI);

// ---------- 底层SPI读写 ----------
static void tft_write_cmd(uint8_t cmd) {
    digitalWrite(TFT_DC, LOW);
    digitalWrite(TFT_CS, LOW);
    tft_spi.write(cmd);
    digitalWrite(TFT_CS, HIGH);
}

static void tft_write_data(uint8_t data) {
    digitalWrite(TFT_DC, HIGH);
    digitalWrite(TFT_CS, LOW);
    tft_spi.write(data);
    digitalWrite(TFT_CS, HIGH);
}

static void tft_write_bulk(const uint8_t* data, size_t len) {
    digitalWrite(TFT_DC, HIGH);
    digitalWrite(TFT_CS, LOW);
    tft_spi.writeBytes(data, len);
    digitalWrite(TFT_CS, HIGH);
}

// ---------- 屏幕初始化 ----------
static void tft_init() {
    pinMode(TFT_CS, OUTPUT); digitalWrite(TFT_CS, HIGH);
    pinMode(TFT_DC, OUTPUT);
    if (TFT_RST >= 0) {
        pinMode(TFT_RST, OUTPUT);
        digitalWrite(TFT_RST, LOW); delay(10);
        digitalWrite(TFT_RST, HIGH); delay(120);
    }
    
    tft_spi.begin(TFT_SCLK, -1, TFT_MOSI, TFT_CS);
    tft_spi.setFrequency(40000000);  // 40MHz SPI
    
    // ILI9341初始化序列
    tft_write_cmd(ILI9341_SWRESET); delay(120);
    tft_write_cmd(ILI9341_SLPOUT);  delay(120);
    
    // 颜色模式: RGB565 16位
    tft_write_cmd(ILI9341_COLMOD);
    tft_write_data(0x55);
    delay(10);
    
    // 内存访问控制: BGR顺序
    tft_write_cmd(ILI9341_MADCTL);
    tft_write_data(0x48);
    
    // 显示开启
    tft_write_cmd(ILI9341_DISPON); delay(120);
    
    // 清屏
    tft_fill_screen(UI_BG);
}

// ---------- 区域设定 ----------
static void tft_set_window(uint16_t x, uint16_t y, uint16_t w, uint16_t h) {
    tft_write_cmd(ILI9341_CASET);
    tft_spi.beginTransaction(SPISettings(40000000, MSBFIRST, SPI_MODE0));
    digitalWrite(TFT_DC, HIGH);
    digitalWrite(TFT_CS, LOW);
    uint8_t data[4] = {(uint8_t)(x>>8), (uint8_t)x, (uint8_t)((x+w-1)>>8), (uint8_t)(x+w-1)};
    tft_spi.writeBytes(data, 4);
    digitalWrite(TFT_CS, HIGH);
    tft_spi.endTransaction();
    
    tft_write_cmd(ILI9341_PASET);
    tft_spi.beginTransaction(SPISettings(40000000, MSBFIRST, SPI_MODE0));
    digitalWrite(TFT_DC, HIGH);
    digitalWrite(TFT_CS, LOW);
    uint8_t data2[4] = {(uint8_t)(y>>8), (uint8_t)y, (uint8_t)((y+h-1)>>8), (uint8_t)(y+h-1)};
    tft_spi.writeBytes(data2, 4);
    digitalWrite(TFT_CS, HIGH);
    tft_spi.endTransaction();
    
    tft_write_cmd(ILI9341_RAMWR);
}

// 填充像素
static void tft_push_color(uint16_t color) {
    tft_spi.beginTransaction(SPISettings(40000000, MSBFIRST, SPI_MODE0));
    digitalWrite(TFT_DC, HIGH);
    digitalWrite(TFT_CS, LOW);
    uint8_t d[2] = {(uint8_t)(color>>8), (uint8_t)color};
    tft_spi.writeBytes(d, 2);
    digitalWrite(TFT_CS, HIGH);
    tft_spi.endTransaction();
}

// ---------- 图形绘制 ----------
static void tft_fill_screen(uint16_t color) {
    tft_set_window(0, 0, 320, 240);
    uint8_t hi = color >> 8, lo = color & 0xFF;
    tft_spi.beginTransaction(SPISettings(40000000, MSBFIRST, SPI_MODE0));
    digitalWrite(TFT_DC, HIGH);
    digitalWrite(TFT_CS, LOW);
    for (int i = 0; i < 320 * 240; i++) {
        tft_spi.write(hi);
        tft_spi.write(lo);
    }
    digitalWrite(TFT_CS, HIGH);
    tft_spi.endTransaction();
}

static void tft_fill_rect(uint16_t x, uint16_t y, uint16_t w, uint16_t h, uint16_t color) {
    tft_set_window(x, y, w, h);
    uint32_t pixels = w * h;
    uint8_t hi = color >> 8, lo = color & 0xFF;
    tft_spi.beginTransaction(SPISettings(40000000, MSBFIRST, SPI_MODE0));
    digitalWrite(TFT_DC, HIGH);
    digitalWrite(TFT_CS, LOW);
    for (uint32_t i = 0; i < pixels; i++) {
        tft_spi.write(hi);
        tft_spi.write(lo);
    }
    digitalWrite(TFT_CS, HIGH);
    tft_spi.endTransaction();
}

static void tft_draw_pixel(uint16_t x, uint16_t y, uint16_t color) {
    tft_set_window(x, y, 1, 1);
    tft_push_color(color);
}

static void tft_draw_rect(uint16_t x, uint16_t y, uint16_t w, uint16_t h, uint16_t color) {
    tft_fill_rect(x, y, w, 1, color);         // 上边
    tft_fill_rect(x, y+h-1, w, 1, color);     // 下边
    tft_fill_rect(x, y, 1, h, color);         // 左边
    tft_fill_rect(x+w-1, y, 1, h, color);     // 右边
}

static void tft_draw_round_rect(uint16_t x, uint16_t y, uint16_t w, uint16_t h, 
    uint16_t r, uint16_t color) {
    // 简化版：只画4个角弧+4条边
    // 实际可用更完善的算法
    tft_draw_rect(x+r, y, w-2*r, 1, color);      // 上
    tft_draw_rect(x+r, y+h-1, w-2*r, 1, color);  // 下
    tft_draw_rect(x, y+r, 1, h-2*r, color);       // 左
    tft_draw_rect(x+w-1, y+r, 1, h-2*r, color);   // 右
}

static void tft_fill_round_rect(uint16_t x, uint16_t y, uint16_t w, uint16_t h,
    uint16_t r, uint16_t color) {
    // 简化版：填充主体+边角
    tft_fill_rect(x+r, y, w-2*r, h, color);    // 主体
    tft_fill_rect(x, y+r, r, h-2*r, color);    // 左
    tft_fill_rect(x+w-r, y+r, r, h-2*r, color);// 右
}

// ---------- 字符/文字显示(简单5x7点阵) ----------
// 本项目使用U8g2中文字体，此处为占位方案
// 正式版建议使用TFT_eSPI库自带字体或U8g2_for_TFT_eSPI

static void tft_draw_char(uint16_t x, uint16_t y, char c, uint16_t color) {
    if (c < 0x20 || c > 0x7E) return;
    // 简单5x7点阵字体
    static const uint8_t font5x7[96][5] = {
        {0x00,0x00,0x00,0x00,0x00}, // space
        {0x00,0x00,0x5F,0x00,0x00}, // !
        {0x00,0x07,0x00,0x07,0x00}, // "
        {0x14,0x7F,0x14,0x7F,0x14}, // #
        {0x24,0x2A,0x7F,0x2A,0x12}, // $
        {0x23,0x13,0x08,0x64,0x62}, // %
        {0x36,0x49,0x55,0x22,0x50}, // &
        {0x00,0x05,0x03,0x00,0x00}, // '
        {0x00,0x1C,0x22,0x41,0x00}, // (
        {0x00,0x41,0x22,0x1C,0x00}, // )
        {0x08,0x2A,0x1C,0x2A,0x08}, // *
        {0x08,0x08,0x3E,0x08,0x08}, // +
        {0x00,0x50,0x30,0x00,0x00}, // ,
        {0x08,0x08,0x08,0x08,0x08}, // -
        {0x00,0x60,0x60,0x00,0x00}, // .
        {0x20,0x10,0x08,0x04,0x02}, // /
    };
    uint8_t idx = c - 0x20;
    if (idx >= sizeof(font5x7)/5) return;
    const uint8_t* glyph = font5x7[idx];
    for (int col = 0; col < 5; col++) {
        for (int row = 0; row < 7; row++) {
            if (glyph[col] & (1 << row))
                tft_draw_pixel(x + col, y - row, color);
        }
    }
}

static void tft_draw_text(uint16_t x, uint16_t y, const char* s, uint16_t color) {
    while (*s) {
        tft_draw_char(x, y, *s, color);
        x += 6;
        if (x > 310) { x = 0; y += 10; }
        s++;
    }
}

// ---------- 卡片绘制 ----------
static void tft_draw_card(int x, int y, int w, int h, 
    uint16_t color, const char* value, const char* label) {
    // 填充底色
    tft_fill_round_rect(x, y, w, h, 4, color);
    // 边框高亮
    tft_draw_round_rect(x, y, w, h, 4, TFT_WHITE);
    // 数值（大号模拟：每字符7x10偏移）
    tft_draw_text(x + 8, y + 25, value, TFT_WHITE);
    // 标签
    tft_draw_text(x + 8, y + 50, label, 0xAD55);  // 灰色
}

// ---------- 迷你曲线 ----------
static void tft_draw_sparkline(int x, int y, int w, int h,
    const float* data, int len, uint16_t color) {
    if (len < 2) return;
    // 计算范围
    float min_v = data[0], max_v = data[0];
    for (int i = 1; i < len; i++) {
        if (data[i] < min_v) min_v = data[i];
        if (data[i] > max_v) max_v = data[i];
    }
    float range = max_v - min_v;
    if (range < 0.1f) range = 0.1f;
    // 绘制折线
    for (int i = 0; i < len - 1; i++) {
        int x1 = x + i * w / (len - 1);
        int x2 = x + (i + 1) * w / (len - 1);
        int y1 = y + h - (int)((data[i] - min_v) / range * (h - 1));
        int y2 = y + h - (int)((data[i+1] - min_v) / range * (h - 1));
        // 画线(DDA简化)
        int dx = abs(x2 - x1), sx = x1 < x2 ? 1 : -1;
        int dy = -abs(y2 - y1), sy = y1 < y2 ? 1 : -1;
        int err = dx + dy;
        int cx = x1, cy = y1;
        while (true) {
            tft_draw_pixel(cx, cy, color);
            if (cx == x2 && cy == y2) break;
            int e2 = 2 * err;
            if (e2 >= dy) { err += dy; cx += sx; }
            if (e2 <= dx) { err += dx; cy += sy; }
        }
    }
}

// ====== 便捷更新接口 ======
static TftPage tft_page;
static unsigned long tft_last_update = 0;
#define TFT_UPDATE_MS 2000  // 2秒刷新一次

static void tft_update(float temp, float humi, uint8_t people, float noise,
    const char* net, bool sentry, const char* asr, unsigned long uptime,
    size_t heap, size_t psram, unsigned long tx) {
    
    tft_page.temp = temp;
    tft_page.humi = humi;
    tft_page.people = people;
    tft_page.noise = noise;
    tft_page.net_status = net;
    tft_page.sentry_on = sentry;
    tft_page.asr_text = asr;
    tft_page.uptime = uptime;
    tft_page.free_heap = heap;
    tft_page.free_psram = psram;
    tft_page.tx_bytes = tx;
    
    unsigned long now = millis();
    if (now - tft_last_update < TFT_UPDATE_MS) return;
    tft_last_update = now;
    
    tft_render_main(tft_page);
}

// 🦞 IPS屏专用测试固件 v1.1 — 全诊断输出
// 只有TFT显示，不初始化其他外设
// 每步都打印到串口，方便定位卡在哪
#include <Arduino.h>
#include <SPI.h>
#include <TFT_eSPI.h>
#include <math.h>
#include <esp_chip_info.h>
#include <esp_flash.h>

TFT_eSPI tft = TFT_eSPI();

// LED诊断编码（不依赖串口，肉眼可见）
// 2次快闪=串口OK  3次=TFT失败  4次=PSRAM失败  5次=帧缓冲失败  长亮=全部正常
#define LED_PIN 2  // ESP32-S3开发板通常GPIO2有板载LED
#define BL_PIN 19  // 屏背光（强制高电平点亮）
void blinkCode(int count, int ms=200) {
  for(int i=0;i<count;i++){ digitalWrite(LED_PIN, HIGH); delay(ms); digitalWrite(LED_PIN, LOW); delay(ms); }
}

void log(const char* msg) {
  Serial.print("[DIAG] "); Serial.println(msg);
}

void logVal(const char* label, unsigned long val) {
  Serial.print("[DIAG] "); Serial.print(label); Serial.print(": "); Serial.println(val);
}

void setup() {
  // 第一步：开启串口
  Serial.begin(115200);
  delay(2000);  // 等USB枚举完成
  Serial.println("\n\n========================================");
  Serial.println("  🦞 IPS屏诊断固件 v1.1");
  Serial.println("========================================");

  // LED编码：2次快闪 = 串口/芯片启动OK
  pinMode(LED_PIN, OUTPUT);
  pinMode(BL_PIN, OUTPUT);
  digitalWrite(BL_PIN, HIGH);  // 强制开背光
  blinkCode(2, 200);
  digitalWrite(LED_PIN, HIGH);  // 进入诊断流程常亮

  // 第二步：芯片信息
  log("--- [1/7] 芯片信息 ---");
  esp_chip_info_t ci;
  esp_chip_info(&ci);
  Serial.printf("  型号: %d 核: %d 版本: %d\n", ci.model, ci.cores, ci.revision);
  Serial.printf("  特性: %s%s%s\n",
    (ci.features & CHIP_FEATURE_WIFI_BGN) ? "WiFi " : "",
    (ci.features & CHIP_FEATURE_BT) ? "BT " : "",
    (ci.features & CHIP_FEATURE_BLE) ? "BLE " : "");
  
  uint32_t flash_size = 0;
  if (esp_flash_get_size(NULL, &flash_size) == ESP_OK) {
    Serial.printf("  Flash: %u MB\n", flash_size / (1024*1024));
  } else {
    log("  Flash: 读取失败!");
  }
  
  logVal("  Heap: ", ESP.getHeapSize());
  logVal("  Free Heap: ", ESP.getFreeHeap());

  // 第三步：PSRAM检查
  log("--- [2/7] PSRAM检查 ---");
  if (psramFound()) {
    Serial.printf("  PSRAM: 总 %u KB 空闲 %u KB\n", 
      ESP.getPsramSize()/1024, ESP.getFreePsram()/1024);
  } else {
    log("  PSRAM: 未检测到! (若无PSRAM需改配置)");
    blinkCode(4, 400);  // 4次慢闪 = 无PSRAM（不致命，继续）
  }

  // 第四步：初始化SPI
  log("--- [3/7] SPI初始化 ---");
  SPI.begin();
  log("  SPI.begin() OK");

  // 第五步：初始化TFT
  log("--- [4/7] TFT初始化 ---");
  log("  引脚: CS=10 DC=11 RST=12 MOSI=13 SCLK=15");
  log("  驱动: ILI9341 320x240 SPI 40MHz");
  log("  开始 tft.init()...");
  
  tft.init();
  log("  tft.init() OK");
  
  tft.setRotation(1);
  log("  setRotation(1) OK");
  
  tft.fillScreen(TFT_BLACK);
  log("  fillScreen(BLACK) OK");

  // 第六步：尝试PSRAM帧缓冲
  log("--- [5/7] PSRAM帧缓冲 ---");
  if (psramFound()) {
    log("  分配 320x240x2=150KB 帧缓冲...");
    uint16_t* fb = (uint16_t*)ps_malloc(320 * 240 * 2);
    if (fb) {
      memset(fb, 0, 320 * 240 * 2);
      tft.setSwapBytes(true);
      log("  帧缓冲 OK");
    } else {
      log("  帧缓冲分配失败（直接用tft绘制）");
      blinkCode(5, 300);  // 5次 = 帧缓冲分配失败（不致命）
    }
  } else {
    log("  无PSRAM，直接用tft绘制");
  }

  // 第七步：显示启动画面
  log("--- [6/7] 显示启动画面 ---");
  tft.setTextColor(TFT_WHITE, TFT_BLACK);
  tft.setTextSize(1);
  tft.drawCentreString("虾兵 IPS 诊断", 160, 60, 2);
  tft.drawCentreString("启动成功", 160, 100, 2);
  tft.drawCentreString("5秒后开始轮播", 160, 140, 1);
  
  log("  启动画面显示OK");
  log("--- [7/7] 启动完成! 5秒后进入UI轮播 ---");
  Serial.println("========================================\n");
  
  // LED长亮 = 全部正常
  digitalWrite(LED_PIN, HIGH);
  
  delay(5000);
}

// 🦞 虾兵 · WiFi CSI 人体检测模块实现

#include "csi_test.h"
#include <WiFi.h>

CSIManager csi_manager;

// ====== CSI 回调（IRAM安全） ======
void IRAM_ATTR CSIManager::csi_callback(void *ctx, wifi_csi_info_t *data) {
    if (!data || !data->buf) return;
    CSIManager *self = static_cast<CSIManager*>(ctx);
    if (self) self->process_csi_packet(data);
}

// ====== 初始化 ======
bool CSIManager::begin() {
    // WiFi必须先初始化并连接
    wifi_mode_t mode;
    esp_wifi_get_mode(&mode);
    if (mode != WIFI_MODE_STA && mode != WIFI_MODE_APSTA) {
        log_w("CSI需要WiFi STA模式，当前模式=%d", mode);
        return false;
    }
    return true;
}

// ====== 启用CSI ======
bool CSIManager::enable() {
    if (enabled_) return true;
    if (!configured_) {
        // 配置CSI参数
        wifi_csi_config_t cfg = {};
        cfg.lltf_en = 1;      // 长训练序列
        cfg.htltf_en = 1;     // HT长训练序列  
        cfg.stbc_htltf2_en = 1;
        cfg.ltf_merge_en = 1;
        cfg.channel_filter_en = 0;
        cfg.manu_scale = 0;
        cfg.shift = 0;
        
        esp_err_t err = esp_wifi_set_csi_config(&cfg);
        if (err != ESP_OK) {
            log_e("CSI配置失败: %s", esp_err_to_name(err));
            return false;
        }
        configured_ = true;
    }
    
    // 关闭WiFi省电模式（必须，否则收不到CSI包）
    esp_wifi_set_ps(WIFI_PS_NONE);
    
    // 注册回调
    esp_err_t err = esp_wifi_set_csi_rx_cb(csi_callback, this);
    if (err != ESP_OK) {
        log_e("CSI注册回调失败: %s", esp_err_to_name(err));
        return false;
    }
    
    // 启用CSI
    err = esp_wifi_set_csi(true);
    if (err != ESP_OK) {
        log_e("CSI启用失败: %s", esp_err_to_name(err));
        return false;
    }
    
    enabled_ = true;
    traffic_timer_ = millis();
    log_i("CSI已启用");
    return true;
}

// ====== 禁用CSI ======
void CSIManager::disable() {
    if (!enabled_) return;
    esp_wifi_set_csi(false);
    enabled_ = false;
}

// ====== 处理CSI数据包 ======
void CSIManager::process_csi_packet(const wifi_csi_info_t *data) {
    if (!data || !data->buf) return;
    
    // 提取空间湍流
    float turb = extract_spatial_turbulence(data);
    
    // 存入环形缓冲区
    turbulence_buf_[buf_head_] = turb;
    buf_head_ = (buf_head_ + 1) % BUF_SIZE;
    if (buf_count_ < BUF_SIZE) buf_count_++;
}

// ====== 提取空间湍流 ======
float CSIManager::extract_spatial_turbulence(const wifi_csi_info_t *data) {
    // CSI数据格式: [Q0, I0, Q1, I1, ...]
    int16_t *csi = (int16_t*)data->buf;
    uint16_t len = data->len / 2;  // 复数个数
    
    int sc_count = (len > 64) ? 64 : len;
    float amps[64] = {0};
    
    for (int i = 0; i < sc_count; i++) {
        float i_val = csi[i * 2 + 1];
        float q_val = csi[i * 2];
        amps[i] = sqrt(i_val * i_val + q_val * q_val);
    }
    
    // 计算所选子载波的均值和标准差
    float sum = 0, sum_sq = 0;
    int n = CSI_SELECTED_COUNT;
    for (int i = 0; i < n; i++) {
        int idx = selected_sc_[i];
        if (idx < sc_count) {
            sum += amps[idx];
            sum_sq += amps[idx] * amps[idx];
        }
    }
    
    float mean = sum / n;
    float variance = sum_sq / n - mean * mean;
    return (variance > 0) ? sqrt(variance) : 0;
}

// ====== 计算移动方差 ======
float CSIManager::calculate_moving_variance() {
    int n = (buf_count_ < CSI_MV_WINDOW) ? buf_count_ : CSI_MV_WINDOW;
    if (n < 5) return 0;
    
    float sum = 0, sum_sq = 0;
    int idx = (buf_head_ - n + BUF_SIZE) % BUF_SIZE;
    for (int i = 0; i < n; i++) {
        float v = turbulence_buf_[(idx + i) % BUF_SIZE];
        sum += v;
        sum_sq += v * v;
    }
    float mean = sum / n;
    return sum_sq / n - mean * mean;  // 方差
}

// ====== 流量发生器（产生WiFi包让CSI采样） ======
void CSIManager::generate_traffic() {
    if (millis() - last_ping_ < 100) return;  // 每秒10包
    last_ping_ = millis();
    // 向网关发UDP包产生WiFi流量
    static WiFiUDP udp;
    IPAddress gw = WiFi.gatewayIP();
    if (gw != IPAddress(0,0,0,0)) {
        udp.beginPacket(gw, 12345);
        udp.write((uint8_t)'C'); udp.write((uint8_t)'S'); udp.write((uint8_t)'I');
        udp.endPacket();
    }
}

// ====== 主循环处理 ======
void CSIManager::process() {
    if (!enabled_) return;
    
    // 产生WiFi流量确保CSI有数据采样
    generate_traffic();
    
    if (buf_count_ < CSI_MV_WINDOW) return;
    
    current_mv_ = calculate_moving_variance();
    motion_state_ = (current_mv_ > threshold_);
}

/* miniz.c 3.1.0 - public domain deflate/inflate, zlib-subset, ZIP reading/writing/appending, PNG writing
   See "unlicense" statement at the end of this file.
   Rich Geldreich <richgel99@gmail.com>, last updated Oct. 13, 2013
   Implements RFC 1950: http://www.ietf.org/rfc/rfc1950.txt and RFC 1951: http://www.ietf.org/rfc/rfc1951.txt

   Most API's defined in miniz.c are optional. For example, to disable the archive related functions just define
   MINIZ_NO_ARCHIVE_APIS, or to get rid of all stdio usage define MINIZ_NO_STDIO (see the list below for more macros).

   * Low-level Deflate/Inflate implementation notes:

     Compression: Use the "tdefl" API's. The compressor supports raw, static, and dynamic blocks, lazy or
     greedy parsing, match length filtering, RLE-only, and Huffman-only streams. It performs and compresses
     approximately as well as zlib.

     Decompression: Use the "tinfl" API's. The entire decompressor is implemented as a single function
     coroutine: see tinfl_decompress(). It supports decompression into a 32KB (or larger power of 2) wrapping buffer, or into a memory
     block large enough to hold the entire file.

     The low-level tdefl/tinfl API's do not make any use of dynamic memory allocation.

   * zlib-style API notes:

     miniz.c implements a fairly large subset of zlib. There's enough functionality present for it to be a drop-in
     zlib replacement in many apps:
        The z_stream struct, optional memory allocation callbacks
        deflateInit/deflateInit2/deflate/deflateReset/deflateEnd/deflateBound
        inflateInit/inflateInit2/inflate/inflateReset/inflateEnd
        compress, compress2, compressBound, uncompress
        CRC-32, Adler-32 - Using modern, minimal code size, CPU cache friendly routines.
        Supports raw deflate streams or standard zlib streams with adler-32 checking.

     Limitations:
      The callback API's are not implemented yet. No support for gzip headers or zlib static dictionaries.
      I've tried to closely emulate zlib's various flavors of stream flushing and return status codes, but
      there are no guarantees that miniz.c pulls this off perfectly.

   * PNG writing: See the tdefl_write_image_to_png_file_in_memory() function, originally written by
     Alex Evans. Supports 1-4 bytes/pixel images.

   * ZIP archive API notes:

     The ZIP archive API's where designed with simplicity and efficiency in mind, with just enough abstraction to
     get the job done with minimal fuss. There are simple API's to retrieve file information, read files from
     existing archives, create new archives, append new files to existing archives, or clone archive data from
     one archive to another. It supports archives located in memory or the heap, on disk (using stdio.h),
     or you can specify custom file read/write callbacks.

     - Archive reading: Just call this function to read a single file from a disk archive:

      void *mz_zip_extract_archive_file_to_heap(const char *pZip_filename, const char *pArchive_name,
        size_t *pSize, mz_uint zip_flags);

     For more complex cases, use the "mz_zip_reader" functions. Upon opening an archive, the entire central
     directory is located and read as-is into memory, and subsequent file access only occurs when reading individual files.

     - Archives file scanning: The simple way is to use this function to scan a loaded archive for a specific file:

     int mz_zip_reader_locate_file(mz_zip_archive *pZip, const char *pName, const char *pComment, mz_uint flags);

     The locate operation can optionally check file comments too, which (as one example) can be used to identify
     multiple versions of the same file in an archive. This function uses a simple linear search through the central
     directory, so it's not very fast.

     Alternately, you can iterate through all the files in an archive (using mz_zip_reader_get_num_files()) and
     retrieve detailed info on each file by calling mz_zip_reader_file_stat().

     - Archive creation: Use the "mz_zip_writer" functions. The ZIP writer immediately writes compressed file data
     to disk and builds an exact image of the central directory in memory. The central directory image is written
     all at once at the end of the archive file when the archive is finalized.

     The archive writer can optionally align each file's local header and file data to any power of 2 alignment,
     which can be useful when the archive will be read from optical media. Also, the writer supports placing
     arbitrary data blobs at the very beginning of ZIP archives. Archives written using either feature are still
     readable by any ZIP tool.

     - Archive appending: The simple way to add a single file to an archive is to call this function:

      mz_bool mz_zip_add_mem_to_archive_file_in_place(const char *pZip_filename, const char *pArchive_name,
        const void *pBuf, size_t buf_size, const void *pComment, mz_uint16 comment_size, mz_uint level_and_flags);

     The archive will be created if it doesn't already exist, otherwise it'll be appended to.
     Note the appending is done in-place and is not an atomic operation, so if something goes wrong
     during the operation it's possible the archive could be left without a central directory (although the local
     file headers and file data will be fine, so the archive will be recoverable).

     For more complex archive modification scenarios:
     1. The safest way is to use a mz_zip_reader to read the existing archive, cloning only those bits you want to
     preserve into a new archive using using the mz_zip_writer_add_from_zip_reader() function (which compiles the
     compressed file data as-is). When you're done, delete the old archive and rename the newly written archive, and
     you're done. This is safe but requires a bunch of temporary disk space or heap memory.

     2. Or, you can convert an mz_zip_reader in-place to an mz_zip_writer using mz_zip_writer_init_from_reader(),

     original archive. (This is basically what mz_zip_add_mem_to_archive_file_in_place() does.) There's a

    /* tdefl_init() compression flags logically OR'd together (low 12 bits contain the max. number of probes per dictionary search): */
    /* TDEFL_DEFAULT_MAX_PROBES: The compressor defaults to 128 dictionary probes per dictionary search. 0=Huffman only, 1=Huffman+LZ (fastest/crap compression), 4095=Huffman+LZ (slowest/best compression). */
    enum
    {
        TDEFL_HUFFMAN_ONLY = 0,
        TDEFL_DEFAULT_MAX_PROBES = 128,
        TDEFL_MAX_PROBES_MASK = 0xFFF
    };

    /* TDEFL_WRITE_ZLIB_HEADER: If set, the compressor outputs a zlib header before the deflate data, and the Adler-32 of the source data at the end. Otherwise, you'll get raw deflate data. */
    /* TDEFL_COMPUTE_ADLER32: Always compute the adler-32 of the input data (even when not writing zlib headers). */
    /* TDEFL_GREEDY_PARSING_FLAG: Set to use faster greedy parsing, instead of more efficient lazy parsing. */
    /* TDEFL_NONDETERMINISTIC_PARSING_FLAG: Enable to decrease the compressor's initialization time to the minimum, but the output may vary from run to run given the same input (depending on the contents of memory). */
    /* TDEFL_RLE_MATCHES: Only look for RLE matches (matches with a distance of 1) */
    /* TDEFL_FILTER_MATCHES: Discards matches <= 5 chars if enabled. */
    /* TDEFL_FORCE_ALL_STATIC_BLOCKS: Disable usage of optimized Huffman tables. */
    /* TDEFL_FORCE_ALL_RAW_BLOCKS: Only use raw (uncompressed) deflate blocks. */
    /* The low 12 bits are reserved to control the max # of hash probes per dictionary lookup (see TDEFL_MAX_PROBES_MASK). */
    enum
    {
        TDEFL_WRITE_ZLIB_HEADER = 0x01000,
        TDEFL_COMPUTE_ADLER32 = 0x02000,
        TDEFL_GREEDY_PARSING_FLAG = 0x04000,
        TDEFL_NONDETERMINISTIC_PARSING_FLAG = 0x08000,
        TDEFL_RLE_MATCHES = 0x10000,
        TDEFL_FILTER_MATCHES = 0x20000,
        TDEFL_FORCE_ALL_STATIC_BLOCKS = 0x40000,
        TDEFL_FORCE_ALL_RAW_BLOCKS = 0x80000
    };

    /* Return status. */
    typedef enum
    {
        /* This flags indicates the inflator needs 1 or more input bytes to make forward progress, but the caller is indicating that no more are available. The compressed data */
        /* is probably corrupted. If you call the inflator again with more bytes it'll try to continue processing the input but this is a BAD sign (either the data is corrupted or you called it incorrectly). */
        /* If you call it again with no input you'll just get TINFL_STATUS_FAILED_CANNOT_MAKE_PROGRESS again. */
        TINFL_STATUS_FAILED_CANNOT_MAKE_PROGRESS = -4,

        /* This flag indicates that one or more of the input parameters was obviously bogus. (You can try calling it again, but if you get this error the calling code is wrong.) */
        TINFL_STATUS_BAD_PARAM = -3,

        /* This flags indicate the inflator is finished but the adler32 check of the uncompressed data didn't match. If you call it again it'll return TINFL_STATUS_DONE. */
        TINFL_STATUS_ADLER32_MISMATCH = -2,

        /* This flags indicate the inflator has somehow failed (bad code, corrupted input, etc.). If you call it again without resetting via tinfl_init() it it'll just keep on returning the same status failure code. */
        TINFL_STATUS_FAILED = -1,

        /* Any status code less than TINFL_STATUS_DONE must indicate a failure. */

        /* This flag indicates the inflator has returned every byte of uncompressed data that it can, has consumed every byte that it needed, has successfully reached the end of the deflate stream, and */
        /* if zlib headers and adler32 checking enabled that it has successfully checked the uncompressed data's adler32. If you call it again you'll just get TINFL_STATUS_DONE over and over again. */
        TINFL_STATUS_DONE = 0,

        /* This flag indicates the inflator MUST have more input data (even 1 byte) before it can make any more forward progress, or you need to clear the TINFL_FLAG_HAS_MORE_INPUT */
        /* flag on the next call if you don't have any more source data. If the source data was somehow corrupted it's also possible (but unlikely) for the inflator to keep on demanding input to */
        /* proceed, so be sure to properly set the TINFL_FLAG_HAS_MORE_INPUT flag. */
        TINFL_STATUS_NEEDS_MORE_INPUT = 1,

        /* This flag indicates the inflator definitely has 1 or more bytes of uncompressed data available, but it cannot write this data into the output buffer. */
        /* Note if the source compressed data was corrupted it's possible for the inflator to return a lot of uncompressed data to the caller. I've been assuming you know how much uncompressed data to expect */
        /* (either exact or worst case) and will stop calling the inflator and fail after receiving too much. In pure streaming scenarios where you have no idea how many bytes to expect this may not be possible */
        /* so I may need to add some code to address this. */
        TINFL_STATUS_HAS_MORE_OUTPUT = 2
    } tinfl_status;

static int f9_find(uint16_t code) {
  if (code >= 0x20 && code <= 0x7E) return code - 0x20 + 1;
  for (int i = 0; i < F9_UNI_N; i++) if (f9_uni[i] == code) return 96 + i;
  return 0;
}

template<typename T>
void f9_draw(T* d, int x, int y, const char* text) {
  while (*text) {
    uint16_t code = (unsigned char)*text++;
    if (code >= 0x80) {
      if ((code & 0xE0) == 0xC0) code = (code & 0x1F) << 6 | ((unsigned char)*text++ & 0x3F);
      else if ((code & 0xF0) == 0xE0) code = (code & 0x0F) << 12 | (((unsigned char)*text++ & 0x3F) << 6) | ((unsigned char)*text++ & 0x3F);
    }
    int g = f9_find(code);
    if (g > 0 && g < 107) {
      int bi = f9_glyph[g][0], bw = f9_glyph[g][2], bh = f9_glyph[g][3], ox = f9_glyph[g][4], oy = f9_glyph[g][5];
      oy = -oy;
      for (int row = 0; row < bh && y + row + oy < 64; row++) {
        for (int col = 0; col < bw && x + col + ox < 128; col++) {
          int idx = bi + row * bw + col;
          if (f9_bits[idx / 8] & (1 << (7 - (idx % 8)))) d->drawPixel(x + col + ox, y + row + oy);
        }
      }
    }
    x += 8;
  }
}
