// Pins: RST=48 CS=45 DC=47 MOSI=21 SCK=20

#include <Arduino.h>

#define PIN_CS    45
#define PIN_DC    47
#define PIN_RST   48
#define PIN_MOSI  21
#define PIN_SCLK  20

#define W 320
#define H 480

// ==================== SPI ====================
static void spi_byte(uint8_t d) {
    for (int i = 0; i < 8; i++) {
        digitalWrite(PIN_SCLK, LOW);
        digitalWrite(PIN_MOSI, (d & 0x80) ? HIGH : LOW);
        d <<= 1;
        digitalWrite(PIN_SCLK, HIGH);
    }
}

static void wr_cmd(uint8_t c) {
    digitalWrite(PIN_DC, LOW);
    digitalWrite(PIN_CS, LOW);
    spi_byte(c);
    digitalWrite(PIN_CS, HIGH);
}

static void wr_data(uint8_t d) {
    digitalWrite(PIN_DC, HIGH);
    digitalWrite(PIN_CS, LOW);
    spi_byte(d);
    digitalWrite(PIN_CS, HIGH);
}

// ==================== 绘制 ====================
void set_win(int x0, int y0, int x1, int y1) {
    if (x0 < 0) x0 = 0; if (y0 < 0) y0 = 0;
    if (x1 >= W) x1 = W-1; if (y1 >= H) y1 = H-1;
    wr_cmd(0x2A);
    wr_data(x0 >> 8); wr_data(x0 & 0xFF);
    wr_data(x1 >> 8); wr_data(x1 & 0xFF);
    wr_cmd(0x2B);
    wr_data(y0 >> 8); wr_data(y0 & 0xFF);
    wr_data(y1 >> 8); wr_data(y1 & 0xFF);
    wr_cmd(0x2C);
}

void fill(int x, int y, int w, int h, uint8_t r, uint8_t g, uint8_t b) {
    if (w <= 0 || h <= 0 || x >= W || y >= H) return;
    if (x < 0) { w += x; x = 0; }
    if (y < 0) { h += y; y = 0; }
    if (x + w > W) w = W - x;
    if (y + h > H) h = H - y;
    set_win(x, y, x + w - 1, y + h - 1);
    digitalWrite(PIN_DC, HIGH);
    digitalWrite(PIN_CS, LOW);
    int n = w * h;
    for (int i = 0; i < n; i++) { spi_byte(r); spi_byte(g); spi_byte(b); }
    digitalWrite(PIN_CS, HIGH);
}

void pixel(int x, int y, uint8_t r, uint8_t g, uint8_t b) {
    if (x < 0 || x >= W || y < 0 || y >= H) return;
    set_win(x, y, x, y);
    digitalWrite(PIN_DC, HIGH);
    digitalWrite(PIN_CS, LOW);
    spi_byte(r); spi_byte(g); spi_byte(b);
    digitalWrite(PIN_CS, HIGH);
}

// ==================== 中文绘制函数 ====================
// 按编码顺序查找字符索引
static int find_cn_idx(char c1, char c2) {
    // 编码字符串: 今偏全出噪在境外头守安定已常度态急感房护按探控日时标正深温湿烟状环电监目省睡知磁离移空紧红线能舒节设调适部钮长门间防音
    static const char* idx_str = "今偏全出噪在境外头守安定已常度态急感房护按探控日时标正深温湿烟状环电监目省睡知磁离移空紧红线能舒节设调适部钮长门间防音";
    const char* p = idx_str;
    int i = 0;
    while (*p) {
        if (*p == c1 && *(p+1) == c2) return i;
        p += 3;  // UTF-8中文字符3字节
        i++;
    }
    return -1;
}

void cn_text(int x, int y, const char* s, uint8_t r, uint8_t g, uint8_t b) {
    while (*s) {
        // 检测UTF-8中文字符 (3字节: 0xE4-0xE9开头)
        if ((*s & 0xE0) == 0xE0 && *(s+1) && *(s+2)) {
            int idx = find_cn_idx(*s, *(s+1));
            if (idx >= 0) {
                int data_idx = idx * 32;
                for (int row = 0; row < 16; row++) {
                    uint8_t bh = pgm_read_byte(&cn16[data_idx + row*2]);
                    uint8_t bl = pgm_read_byte(&cn16[data_idx + row*2 + 1]);
                    for (int col = 0; col < 8; col++) {
                        if (bh & (0x80 >> col)) pixel(x + col, y + row, r, g, b);
                        if (bl & (0x80 >> col)) pixel(x + col + 8, y + row, r, g, b);
                    }
                }
                x += 18;  // 16px宽 + 2px间距
            }
            s += 3;
        } else {
            // ASCII字符走原来逻辑
            char ch = *s;
            if (ch >= 32 && ch <= 126) {
                int idx2 = (ch - 32) * 5;
                for (int col = 0; col < 5; col++) {
                    uint8_t bits = pgm_read_byte(&f5x7[idx2 + col]);
                    for (int row = 0; row < 7; row++) {
                        if (bits & (1 << row))
                            pixel(x + col, y + 6 - row, r, g, b);
                    }
                }
                x += 6;
            }
            s++;
        }
    }
}
