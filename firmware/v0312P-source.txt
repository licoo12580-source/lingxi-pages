// 硬件: ESP32-S3 + CL40BC299-40A (ILI9488, SPI 4线, 320x480)
// 引脚: RST=48 CS=45 DC=47 MOSI=21 SCK=20

#include <Arduino.h>

// ==================== 引脚定义 ====================
#define PIN_CS    45
#define PIN_DC    47
#define PIN_RST   48
#define PIN_MOSI  21
#define PIN_SCLK  20

#define WIDTH   320
#define HEIGHT  480

// ==================== 颜色（RGB666 18位，6bit/ch） ====================
#define C_BLACK   0x00, 0x00, 0x00
#define C_WHITE   0xFC, 0xFC, 0xFC

#define C_SOFT    0x80, 0xA0, 0xC0   // 浅蓝

// ==================== SPI ====================
static void spi_byte(uint8_t d) {
    for (int i = 0; i < 8; i++) {
        digitalWrite(PIN_SCLK, LOW);
        digitalWrite(PIN_MOSI, (d & 0x80) ? HIGH : LOW);
        d <<= 1;
        digitalWrite(PIN_SCLK, HIGH);
    }
}

static void wr_cmd(uint8_t c) {
    digitalWrite(PIN_DC, LOW);
    digitalWrite(PIN_CS, LOW);
    spi_byte(c);
    digitalWrite(PIN_CS, HIGH);
}

static void wr_data(uint8_t d) {
    digitalWrite(PIN_DC, HIGH);
    digitalWrite(PIN_CS, LOW);
    spi_byte(d);
    digitalWrite(PIN_CS, HIGH);
}

// 写3字节RGB666像素
static void wr_pixel(uint8_t r, uint8_t g, uint8_t b) {
    digitalWrite(PIN_DC, HIGH);
    digitalWrite(PIN_CS, LOW);
    spi_byte(r);
    spi_byte(g);
    spi_byte(b);
    digitalWrite(PIN_CS, HIGH);
}

// ==================== 区域绘制 ====================

void set_win(uint16_t x0, uint16_t y0, uint16_t x1, uint16_t y1) {
    wr_cmd(0x2A);
    wr_data(x0 >> 8); wr_data(x0 & 0xFF);
    wr_data(x1 >> 8); wr_data(x1 & 0xFF);
    wr_cmd(0x2B);
    wr_data(y0 >> 8); wr_data(y0 & 0xFF);
    wr_data(y1 >> 8); wr_data(y1 & 0xFF);
    wr_cmd(0x2C);
}

void fill_rect(int x, int y, int w, int h, uint8_t r, uint8_t g, uint8_t b) {
    if (x < 0) { w += x; x = 0; }
    if (y < 0) { h += y; y = 0; }
    if (x + w > WIDTH)  w = WIDTH - x;
    if (y + h > HEIGHT) h = HEIGHT - y;
    if (w <= 0 || h <= 0) return;
    set_win(x, y, x + w - 1, y + h - 1);
    digitalWrite(PIN_DC, HIGH);
    digitalWrite(PIN_CS, LOW);
    int n = w * h;
    for (int i = 0; i < n; i++) {
        spi_byte(r); spi_byte(g); spi_byte(b);
    }
    digitalWrite(PIN_CS, HIGH);
}

void draw_char(int x, int y, char c, uint8_t r, uint8_t g, uint8_t b) {
    if (c < 32 || c > 126) c = ' ';
    int idx = (c - 32) * 5;
    for (int col = 0; col < 5; col++) {
        uint8_t bits = pgm_read_byte(&font5x7[idx + col]);
        for (int row = 0; row < 7; row++) {
            if (bits & (0x01 << row)) {
                set_win(x + col, y + (6 - row), x + col, y + (6 - row));
                digitalWrite(PIN_DC, HIGH);
                digitalWrite(PIN_CS, LOW);
                spi_byte(r); spi_byte(g); spi_byte(b);
                digitalWrite(PIN_CS, HIGH);
            }
        }
    }
}

void draw_str(int x, int y, const char* s, uint8_t r, uint8_t g, uint8_t b) {
    // 批量绘制：每行先计算所有字符的像素，然后一整行刷
    // 对于5x7字体，7行，每行宽度=字符数*6
    int len = strlen(s);
    if (len == 0) return;
    
    // 对每个字符，按列逐行绘制
    for (int col = 0; col < 5; col++) {
        // 收集这一列所有字符的像素
        for (int c = 0; c < len; c++) {
            char ch = s[c];
            if (ch < 32 || ch > 126) ch = ' ';
            uint8_t bits = pgm_read_byte(&font5x7[(ch - 32) * 5 + col]);
            
            // 画这一列中所有需要亮的行
            for (int row = 0; row < 7; row++) {
                if (bits & (0x01 << row)) {
                    int px = x + c * 6 + col;
                    int py = y + (6 - row);
                    set_win(px, py, px, py);
                    digitalWrite(PIN_DC, HIGH);
                    digitalWrite(PIN_CS, LOW);
                    spi_byte(r); spi_byte(g); spi_byte(b);
                    digitalWrite(PIN_CS, HIGH);
                }
            }
        }
        yield();
    }
}

// ==================== 主函数 ====================

void setup() {
    Serial.begin(115200);
    delay(2000);
    
    Serial.println("灵犀 v0312P - IPS滚动UI");
    
    pinMode(PIN_CS, OUTPUT);   digitalWrite(PIN_CS, HIGH);
    pinMode(PIN_DC, OUTPUT);   digitalWrite(PIN_DC, HIGH);
    pinMode(PIN_RST, OUTPUT);  digitalWrite(PIN_RST, HIGH);
    pinMode(PIN_MOSI, OUTPUT); digitalWrite(PIN_MOSI, LOW);
    pinMode(PIN_SCLK, OUTPUT); digitalWrite(PIN_SCLK, LOW);
    
    lcd_init();
    Serial.println("屏幕初始化完成");
    
    // 第一页直接画
    fill_rect(0, 0, WIDTH, HEIGHT, C_DARK);
    Serial.println("UI启动");
}

void loop() {
    // 页面循环：每页停留4秒
    while (true) {
        page_room_status(0,0,0,0);
        delay(4500);
        yield();
        
        page_energy(0,0,0,0);
        delay(4500);
        yield();
        
        page_system(0,0,0,0);
        delay(4500);
        yield();
    }
}
